75 float iec_dB_to_fader (float db)
76 {
77  float def = 0.0f; // Meter deflection %age
78 
79  if (db < -70.0f) {
80  def = 0.0f;
81  } else if (db < -60.0f) {
82  def = (db + 70.0f) * 0.25f;
83  } else if (db < -50.0f) {
84  def = (db + 60.0f) * 0.5f + 2.5f; // used to be +5.0f, but this caused a discontinuity in low dB deflections
85  } else if (db < -40.0f) {
86  def = (db + 50.0f) * 0.75f + 7.5f;
87  } else if (db < -30.0f) {
88  def = (db + 40.0f) * 1.5f + 15.0f;
89  } else if (db < -20.0f) {
90  def = (db + 30.0f) * 2.0f + 30.0f;
91  } else {
92  def = (db + 20.0f) * 2.5f + 50.0f;
93  }
94 
95  return def;
96 }

98 static float iec_fader_to_dB(float def) // Meter deflection %age
99 {
100  float db = 0.0f;
101 
102  if (def >= 50.0f) {
103  db = (def - 50.0f) / 2.5f - 20.0f;
104  } else if (def >= 30.0f) {
105  db = (def - 30.0f) / 2.0f - 30.0f;
106  } else if (def >= 15.0f) {
107  db = (def - 15.0f) / 1.5f - 40.0f;
108  } else if (def >= 7.5f) {
109  db = (def - 7.5f) / 0.75f - 50.0f;
110  } else if (def >= 2.5f) { // used to be 0.5, ...
111  db = (def - 2.5f) / 0.5f - 60.0f; // used to be (def - 5.0f), but this caused a discontinuity in low dB deflections
112  } else {
113  db = (def / 0.25f) - 70.0f;
114  }
115 
116  return db;
117 }

130 float AudioLevel::dB_to_fader (float dB, float maxDb)
131 {
132  if (dB == DB_FLOOR)
133  return 0.;
134 
135  // The IEC scale gives a "percentage travel" for a given dB
136  // level, but it reaches 100% at 0dB. So we want to treat the
137  // result not as a percentage, but as a scale between 0 and
138  // whatever the "percentage" for our (possibly >0dB) max dB is.
139 
140  float maxPercent = iec_dB_to_fader(maxDb);
141  float percent = iec_dB_to_fader(dB);
142  float faderLevel = percent / maxPercent;
143 
144  if (faderLevel < 0.) faderLevel = 0.;
145  if (faderLevel > 1.0) faderLevel = 1.0;
146 
147  return faderLevel;
148 }

185 // Mixes the selected inputs (nonzero) to dest
186 // Dest is an array of 2 pointers to left & right buffers
187 // If dest[1] is null, mix is done in mono
188 // Also, we do compensation for DC offset of inputs
189 //
190 // *** Glitch/inefficiency: MixInputs is called for each RecordProcessor
191 // And for PassthroughProcessor- DC offsets are recomputed, etc
192 void AudioBuffers::MixInputs (nframes_t len, sample_t **dest,
193  InputSettings *iset, float inputvol,
194  char compute_stats) {
195  const static int DCOFS_MINIMUM_SAMPLE_COUNT = 10000;
196  const static float DCOFS_LOWPASS_COEFF = 0.99,
197  DCOFS_ONEMINUS_LOWPASS_COEFF = 1.0-DCOFS_LOWPASS_COEFF;
198  const static nframes_t PEAK_HOLD_LENGTH = 1;
199 
200  // Mix together selected inputs..
201  int stereomix = (dest[1] != 0 ? 1 : 0);
202  memset(dest[0],0,sizeof(sample_t) * len);
203  if (stereomix)
204  memset(dest[1],0,sizeof(sample_t) * len);
205 
206  nframes_t phold = 0;
207  if (compute_stats)
208  phold = app->getAUDIO()->get_srate() * PEAK_HOLD_LENGTH;
209 
210  for (int i = 0; i < numins; i++) {
211  if (iset->selins[i]) { // If input is selected!
212  for (int j = 0; j <= stereomix; j++) {
213  // Left & right channels
214  sample_t *in = ins[j][i];
215  float vol = iset->invols[i] * inputvol;
216 
217  // DC offset compute
218  sample_t sum = 0,
219  peak = 0,
220  dcofs = iset->insavg[j][i];
221  nframes_t peaktime = 0;
222  int cnt = 0;
223 
224  if (compute_stats) {
225  cnt = iset->inscnt[i];
226  sum = iset->insums[j][i];
227  peaktime = iset->inpeaktime[i];
228  if (cnt-peaktime > phold)
229  // Old peak- fall to zero
230  peak = 0;
231  else
232  // Retain peak
233  peak = iset->inpeak[i];
234  }
235  if (j == 0) {
236  // Left
237  if (compute_stats) {
238  // Stats
239  for (nframes_t idx = 0; idx < len; idx++) {
240  sample_t s = in[idx],
241  sabs = fabs(s);
242  sum += s;
243  if (sabs > peak) {
244  peak = sabs;
245  peaktime = cnt;
246  }
247  cnt++;
248  dest[0][idx] += (s-dcofs) * vol;
249  }
250  iset->inscnt[i] = cnt;
251  } else {
252  // No stats
253  for (nframes_t idx = 0; idx < len; idx++)
254  dest[0][idx] += (in[idx]-dcofs) * vol;
255  }
256  } else {
257  // If input is mono, take signal from left channel and
258  // use it for the right output
259  if (in == 0)
260  in = ins[0][i];
261 
262  if (compute_stats) {
263  // Stats
264  for (nframes_t idx = 0; idx < len; idx++) {
265  sample_t s = in[idx],
266  sabs = fabs(s);
267  sum += s;
268  if (sabs > peak) {
269  peak = sabs;
270  peaktime = cnt;
271  }
272  dest[j][idx] += (s-dcofs) * vol;
273  }
274  } else {
275  // No stats
276  for (nframes_t idx = 0; idx < len; idx++)
277  dest[j][idx] += (in[idx]-dcofs) * vol;
278  }
279  }
280 
281  // DC offset adjust
282  if (compute_stats) {
283  iset->insums[j][i] = sum;
284  iset->inpeak[i] = peak;
285  iset->inpeaktime[i] = peaktime;
286  if (cnt > DCOFS_MINIMUM_SAMPLE_COUNT)
287  iset->insavg[j][i] =
288  (DCOFS_LOWPASS_COEFF*iset->insavg[j][i]) +
289  (DCOFS_ONEMINUS_LOWPASS_COEFF*sum/cnt);
290  }
291  }
292  }
293  }
294 }

473 void Pulse::process(char pre, nframes_t l, AudioBuffers *ab) {
474  static int midi_clock_count = 0,
475  midi_beat_count = 0;
476 
477  // If we're using Jack transport (audio) sync, and we're slave to another app,
478  // adjust pulse to stay in-sync
479  if (app->getAUDIO()->IsTransportRolling() &&
480  !app->getAUDIO()->IsTimebaseMaster()) {
481  char sync_type = app->GetSyncType();
482  int sync_speed = app->GetSyncSpeed();
483  if (sync_type != prev_sync_type ||
484  sync_speed != prev_sync_speed) {
485  // Make sure we recalculate length / wrap if sync params changed
486  prevbpm = 0;
487  prev_sync_bb = -1;
488 
489  prev_sync_type = sync_type;
490  prev_sync_speed = sync_speed;
491  }
492 
493  double bpm = app->getAUDIO()->GetTransport_BPM();
494  if (bpm != prevbpm) {
495  // Adjust pulse length from BPM
496  float mult = (sync_type ?
497  sync_speed :
498  app->getAUDIO()->GetTransport_BPB()*
499  sync_speed);
500  len = (nframes_t) ((double) 60.*app->getAUDIO()->get_srate()*mult/bpm);
501 
502  // printf("slave to bpm: %lf, len: %d\n",bpm,len);
503  prevbpm = bpm;
504  }
505 
506  int sync_bb = (sync_type ?
507  app->getAUDIO()->GetTransport_Beat() :
508  app->getAUDIO()->GetTransport_Bar());
509  if (sync_bb != prev_sync_bb) {
510  sync_cnt++;
511  if (sync_cnt >= sync_speed) {
512  // Wrap pulse- enough bars or beats has passed
513  sync_cnt = 0;
514  Wrap();
515  }
516 
517  // printf("slave to sync: %d\n",sync_bb);
518  prev_sync_bb = sync_bb;
519  }
520  }
521 
522  nframes_t fragmentsize = app->getBUFSZ();
523  if (l > fragmentsize)
524  l = fragmentsize;
525 
526  // Process on left channel of first output
527  sample_t *out = ab->outs[0][0];
528  nframes_t ofs = 0;
529  if (!pre && !stopped) {
530  nframes_t remaining = len-curpos;
531 
532  // Move forward pulse position
533  wrapped = 0;
534  nframes_t oldpos = curpos;
535  curpos += MIN(l,remaining);
536 
537  // If we are transmitting MIDI sync, do so now
538  if (clockrun != SS_NONE && app->getMIDI()->GetMIDISyncTransmit()) {
539  char sync_type = app->GetSyncType();
540  int sync_speed = app->GetSyncSpeed();
541 
542  // 1 pulse = how many MIDI clock messages to send?
543  // In sync-beat mode, each pulse is SyncSpeed beats, so MIDI_CLOCK_FREQUENCY*SyncSpeed clocks to send
544  // In sync-bar mode, each pulse is one bar, so MIDI_CLOCK_FREQUENCY*BeatsPerBar*SyncSpeed clocks to send
545  int clocksperpulse = MIDI_CLOCK_FREQUENCY*sync_speed;
546  if (!sync_type)
547  clocksperpulse *= SYNC_BEATS_PER_BAR;
548 
549  // Check timing of pulse
550  float framesperclock = (float) len/clocksperpulse;
551  int oldclock = (int) ((float) oldpos/framesperclock),
552  newclock = (int) ((float) curpos/framesperclock);
553  if ((clockrun == SS_BEAT && newclock != oldclock) || curpos >= len) {
554  // printf("CLOCKY-OO %d!\n",newclock);
555 
556  if (clockrun == SS_START) {
557  // Send MIDI start for pulse
558  metrohiofs = 0; // Sound metronome high tone
559  midi_clock_count = 0;
560  midi_beat_count = 0;
561 
562  MIDIStartStopInputEvent *ssevt =
563  (MIDIStartStopInputEvent *) Event::GetEventByType(T_EV_Input_MIDIStartStop);
564  ssevt->start = 1;
565  app->getEMG()->BroadcastEventNow(ssevt, this);
566 
567  // If this is the first downbeat, start the clock proper
568  clockrun = SS_BEAT;
569  } else {
570  midi_clock_count++;
571  if (midi_clock_count >= MIDI_CLOCK_FREQUENCY) {
572  // Quarter not has passed, sound metronome low tone
573  midi_clock_count = 0;
574  // printf("PULSE: quarter clock\n");
575 
576  midi_beat_count++;
577  if (midi_beat_count >= clocksperpulse/MIDI_CLOCK_FREQUENCY) {
578  midi_beat_count = 0;
579  metrohiofs = 0; // Sound metronome high tone
580  } else
581  metroloofs = 0; // Sound metronome low tone
582  }
583 
584  // Time for another clock message
585  MIDIClockInputEvent *clkevt = (MIDIClockInputEvent *) Event::GetEventByType(T_EV_Input_MIDIClock);
586  // *** Broadcast immediately-- this will yield lowest MIDI sync latency,
587  // but may cause problems if MIDI transmit code blocks/conflicts with RT audio thread
588  // (which it very well might)
589  // _experimental_
590  app->getEMG()->BroadcastEventNow(clkevt, this);
591  }
592  }
593  }
594 
595  // Check pulse wrap
596  if (curpos >= len) {
597  // Downbeat!!
598  wrapped = 1;
599  curpos = 0;
600 
601  // Long count
602  lc_cur++;
603  if (lc_cur >= lc_len)
604  lc_cur = 0;
605 
606  // Send out a pulse sync event
607  PulseSyncEvent *pevt = (PulseSyncEvent *)
608  Event::GetEventByType(T_EV_PulseSync);
609  app->getEMG()->BroadcastEventNow(pevt, this);
610 
611  // Trigger management of any RT work on blocks
612  // (currently PulseSync event handles loop points, but
613  // BMG handles striping through StripeBlockManager)
614  // *** To be consolidated into EMG?
615  app->getBMG()->HiPriTrigger(this);
616 
617  // Metronome sound on
618  metroofs = 0;
