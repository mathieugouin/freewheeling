60 CircularMap::CircularMap(SDL_Surface *in,
61  int map_xs, int map_ys,
62  int in_xs, int in_ys,
63  int rinner, int rsize) :
64  in(in), map_xs(map_xs), map_ys(map_ys), in_xs(in_xs), in_ys(in_ys),
65  rinner(rinner), rsize(rsize), next(0) {
66  // Allocate memory for maps
67  map = new Uint8 *[map_xs * map_ys];
68  // 4 scan run lengths per scanline
69  scanmap = new int[4 * map_ys];
70 
71  // Compute constants
72  int map_xc = map_xs/2;
73  int map_yc = map_ys/2;
74  int pitch = in->pitch;
75  int bpp = in->format->BytesPerPixel;
76  Uint8 *in_base = (Uint8 *) in->pixels;
77 
78  // Now generate, interating across output range
79  for (int y = 0; y < map_ys; y++) {
80  int runnum = 0,
81  // Count of consecutive written/notwritten pixels
82  // Positive is written, negative is notwritten
83  pixelscount = 0;
84  for (int x = 0; x < map_xs; x++) {
85  //printf("%d %d\n", x, y);
86  int yofs = y-map_yc,
87  xofs = map_xc-x;
88  float theta = atan2(yofs,xofs),
89  in_x = in_xs*(theta+M_PI)/(2*M_PI);
90 
91  // Now that we know x mapping, based on calculated theta (see theta)
92  // Let's get y mapping
93  float in_y;
94  if (sin(theta) == 0) {
95  // This fixes an annoying horizontal crack in the map
96  in_y = (xofs-rinner)*in_ys/rsize;
97  //printf("xofs: %d yofs: %d inx: %f iny: %f\n",xofs,yofs,in_x,in_y);
98  }
99  else
100  in_y = (yofs/sin(theta)-rinner)*in_ys/rsize;
101 
102  // Are we in range, honey?
103  int idx = y*map_xs + x;
104  if (in_x >= 0 && in_y >= 0 &&
105  in_x < in_xs && in_y < in_ys) {
106  /*printf("%d %d\n", x, y);
107  printf(" in[%d,%d]\n",(int)in_x,(int)in_y);*/
108 
109  // Yup-- write the map
110  map[idx] = in_base + myround(in_y)*pitch + myround(in_x)*bpp;
111 
112  // Generate scan map
113  if (pixelscount <= 0) {
114  // OK start of run
115 
116  // Write number of empty pixels from last run to here
117  scanmap[y*4 + runnum] = -pixelscount;
118  //printf("Y:%d r:%d cnt: %d\n",y,runnum,-pixelscount);
119 
120  runnum++;
121  pixelscount = 0;
122  }
123  pixelscount++;
124  } else {
125  //printf(" none\n");
126 
127  // Mapping is empty for this location
128  map[idx] = 0;
129 
130  // Generate scan map
131  if (pixelscount > 0) {
132  // OK end of run
133 
134  // Write number of pixels written from beginning of run to here
135  scanmap[y*4 + runnum] = pixelscount;
136  //printf("Y:%d r:%d cnt: %d\n",y,runnum,pixelscount);
137 
138  runnum++;
139  pixelscount = 0;
140  }
141  pixelscount--;
142  }
143  }
144 
145  // Now write the rest of scan map
146  if (pixelscount > 0) {
147  // OK end of run
148 
149  // Write number of pixels written from beginning of run to here
150  scanmap[y*4 + runnum] = pixelscount;
151  //printf("Y:%d r:%d cnt: %d\n",y,runnum,pixelscount);
152 
153  runnum++;
154  }
155 
156  //printf("EOL- r:%d\n",runnum);
157  //usleep(100000);
158 
159  for (; runnum < 4; runnum++)
160  scanmap[y*4 + runnum] = -1;
161  }
162 }

171 char CircularMap::Map(SDL_Surface *out, int dstx, int dsty) {
172  int bpp;
173  int *tmpscan = scanmap;
174 
175  // Get surface format
176  bpp = out->format->BytesPerPixel;
177  if (bpp != in->format->BytesPerPixel) {
178  printf("VIDEO: ERROR: Temporary buffer & video screen not "
179  "matching depth (in: %d out: %d).\n",
180  in->format->BytesPerPixel,bpp);
181  return 1;
182  }
183 
184  // Check for clipping
185  if (dstx < 0 || dsty < 0 || dstx+map_xs >= out->w ||
186  dsty+map_ys >= out->h)
187  return 1; // Don't handle clip case-- just don't draw!
188 
189  /*
190  * Lock the surface
191  */
192  if (SDL_MUSTLOCK(out)) {
193  if (SDL_LockSurface(out) < 0) {
194  return 1;
195  }
196  }
197 
198  int out_pitch = out->pitch;
199 
200  Uint8 **ptr = map,
201  **ptr2,
202  *optr = (Uint8 *) out->pixels + dsty*out_pitch + dstx*bpp,
203  *optr2;
204 
205  Uint8 *op, *ip;
206  int ofs,
207  scanleft;
208  for (int y = 0; y < map_ys; y++, ptr += map_xs, optr += out_pitch,
209  tmpscan += scanleft) {
210  scanleft = 4;
211  ptr2 = ptr;
212  optr2 = optr;
213 
214  // What's the first x location on this scanline we should look at?
215  while ((ofs = *tmpscan) != -1 && scanleft) {
216  tmpscan++;
217 
218  // Starting position on this scanline
219  ptr2 += ofs;
220  optr2 += ofs*bpp;
221 
222  //printf("wy:%d sofs:%d ",y,ofs);
223 
224  // And the number of bytes in the run on this scanline
225  ofs = *(tmpscan++);
226 
227  //printf("len:%d\n",ofs);
228 
229  // Depending on the number of bytes per pixel, handle this scanline
230  // differently-- optimized subroutines follow..
231  switch (bpp) {
232  case 1:
233  {
234  for (int i = 0; i < ofs; i++, ptr2++, optr2 += bpp)
235  // Copy 1 byte pixels
236  *optr2 = **ptr2;
237  }
238  break;
239  case 2:
240  {
241  for (int i = 0; i < ofs; i++, ptr2++, optr2 += bpp)
242  // Copy 2 byte pixels
243  *((Uint16 *) optr2) = *((Uint16 *) (*ptr2));
244  }
245  break;
246  case 3:
247  {
248  // 3 byte pixels
249  for (int i = 0; i < ofs; i++, ptr2++, optr2 += bpp) {
250  // Access the map to find offset into input
251  ip = *ptr2;
252  // And output pointer is right there
253  op = optr2;
254 
255  if (SDL_BYTEORDER == SDL_BIG_ENDIAN) {
256  ip += 2;
257  // Copy, reversing bit order for this pixel
258  for (int n = 0; n < bpp; n++)
259  *(op++) = *(ip--);
260  }
261  else {
262  // Copy, regular bit order for this pixel
263  for (int n = 0; n < bpp; n++)
264  *(op++) = *(ip++);
265  }
266  }
267  }
268  break;
269  case 4:
270  {
271  for (int i = 0; i < ofs; i++, ptr2++, optr2 += bpp)
272  // Copy 4 byte pixels
273  *((Uint32 *) optr2) = *((Uint32 *) (*ptr2));
274  }
275  break;
276  }
277 
278  // Ok, 1 scan run complete, so 2 endpoints less on scanline
279  scanleft -= 2;
280  }
281  }
282 
283  /*
284  * Unlock surface
285  */
286  if (SDL_MUSTLOCK(out)) {
287  SDL_UnlockSurface(out);
288  }
289 
290  return 0;
291 };

373  case T_EV_Input_MouseMotion :
374  {
375  MouseMotionInputEvent *mev = (MouseMotionInputEvent *) ev;
376 
377  // Notify all visible browsers of mouse move

379  char skip = 0;

381  if ((curdisplay->show || curdisplay->forceshow) &&
382  curdisplay->GetFloDisplayType() == FD_Browser)
383  skip = ((Browser *) curdisplay)->MouseMotion(mev);

385  curdisplay = curdisplay->next;

389 
390  case T_EV_Input_MouseButton :

392  MouseButtonInputEvent *mev = (MouseButtonInputEvent *) ev;
393 
394  // Notify all visible browsers of button press
395  FloDisplay *curdisplay = app->getCFG()->GetDisplays();
396  char skip = 0;

402  curdisplay = curdisplay->next;

404

622 
623  case T_EV_VideoShowHelp :

654 // This is a custom surface blitter that doesn't use large block
655 // memory writes and thus avoids the strange video glitch of introducing
656 // audio pops on some machines-- some loss in performance!
657 void VideoIO::Custom_BlitSurface(SDL_Surface *in, SDL_Surface *out,
658  SDL_Rect *dstrect) {
659  int opitch = out->pitch,
660  ipitch = in->pitch,
661  bpp = out->format->BytesPerPixel;
662  if (bpp != in->format->BytesPerPixel) {
663  printf("VIDEO: ERROR: Buffer and screen different format!\n");
664  return;
665  }
666 
667  /*
668  * Lock the surface
669  */
670  if (SDL_MUSTLOCK(out)) {
671  if (SDL_LockSurface(out) < 0) {
672  return;
673  }
674  }
675 
676  Uint8 *opixels = (Uint8 *)out->pixels + dstrect->y*opitch + dstrect->x*bpp,
677  *ipixels = (Uint8 *)in->pixels;
678  int xrun = (dstrect->x+dstrect->w <= out->w ? dstrect->w :
679  out->w-dstrect->x),
680  yrun = (dstrect->y+dstrect->h <= out->h ? dstrect->h : out->h-dstrect->y),
681  ojump = opitch-xrun*bpp,
682  ijump = ipitch-xrun*bpp;
683  switch (bpp) {
684  case 1 :
685  {
