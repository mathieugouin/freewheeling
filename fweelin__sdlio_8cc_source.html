596 void *SDLIO::run_sdl_thread(void *ptr)
597 {
598  SDLIO *inst = static_cast<SDLIO *>(ptr);
599  SDL_Event event;
600 
601  printf("SDLIO: SDL Input thread start.\n");
602 
603 #ifdef __MACOSX__
604  // inst->cocoa.SetupCocoaThread();
605 #endif
606 
607  // Setup joysticks
608  inst->numjoy = SDL_NumJoysticks();
609  if (inst->numjoy > 0)
610  inst->joys = new SDL_Joystick *[inst->numjoy];
611  printf("SDLIO: Detected %d joysticks..\n", inst->numjoy);
612  for (int i = 0; i < inst->numjoy; i++) {
613  printf(" Joystick #%d: %s\n",i+1,SDL_JoystickName(i));
614  inst->joys[i] = SDL_JoystickOpen(i);
615  }
616 
617  // Listen for pertinent events
618  if (inst->app->getEMG() == 0) {
619  printf("INIT: Error: Event Manager not yet active!\n");
620  exit(1);
621  }
622  inst->app->getEMG()->ListenEvent(inst,0,T_EV_ExitSession);
623 
624  while (inst->sdlthreadgo) {
625  if (SDL_WaitEvent(&event)) {
626  switch (event.type) {
627  case SDL_JOYBUTTONDOWN :
628  case SDL_JOYBUTTONUP :
629  {
630  JoystickButtonInputEvent *jevt = (JoystickButtonInputEvent *)
631  Event::GetEventByType(T_EV_Input_JoystickButton);
632 
633  jevt->joystick = event.jbutton.which;
634  jevt->button = event.jbutton.button;
635  jevt->down = (event.type == SDL_JOYBUTTONUP ? 0 : 1);
636  inst->app->getEMG()->BroadcastEventNow(jevt, inst);
637 
638  if (inst->app->getCFG()->IsDebugInfo())
639  printf("JOYSTICK: Joystick #%d, button #%d %s\n",
640  jevt->joystick,jevt->button,
641  (jevt->down ? "pressed" : "released"));
642  }
643  break;
644 
645  case SDL_MOUSEMOTION :
646  {
647  MouseMotionInputEvent *mevt = (MouseMotionInputEvent *)
648  Event::GetEventByType(T_EV_Input_MouseMotion);
649 
650  mevt->x = event.motion.x;
651  mevt->y = event.motion.y;
652  inst->app->getEMG()->BroadcastEventNow(mevt, inst);
653 
654  // No debug info for mouse motion
655 #if 0
656  if (inst->app->getCFG()->IsDebugInfo())
657  printf("MOUSE: Motion: (%d,%d)\n",
658  mevt->x, mevt->y);
659 #endif
660  }
661 
662  // printf("x: %d y: %d\n",
663  // event.motion.x,event.motion.y);
664  break;
665 
666  case SDL_MOUSEBUTTONDOWN :
667  case SDL_MOUSEBUTTONUP :
668  {
669  MouseButtonInputEvent *mevt = (MouseButtonInputEvent *)
670  Event::GetEventByType(T_EV_Input_MouseButton);
671 
672  mevt->button = event.button.button;
673  mevt->down = (event.type == SDL_MOUSEBUTTONUP ? 0 : 1);
674  mevt->x = event.button.x;
675  mevt->y = event.button.y;
676  inst->app->getEMG()->BroadcastEventNow(mevt, inst);
677 
678  if (inst->app->getCFG()->IsDebugInfo())
679  printf("MOUSE: Button #%d %s @ (%d,%d)\n",
680  mevt->button,
681  (mevt->down ? "pressed" : "released"),
682  mevt->x, mevt->y);
683  }
684 
685  //printf("button: %d x: %d y: %d\n",
686  // event.button.button,
687  // event.button.x,event.button.y);
688  break;
689 
690  case SDL_KEYDOWN :
691  {
692  SDLKey sym = event.key.keysym.sym;
693  if (sym >= SDLK_FIRST && sym < SDLK_LAST) {
694  // Mark the key as held down
695  inst->keyheld[sym] = 1;
696 
697  // Now generate an input event..
698  KeyInputEvent *kevt = (KeyInputEvent *)
699  Event::GetEventByType(T_EV_Input_Key);
700 
701  kevt->down = 1;
702  kevt->keysym = sym;
703  kevt->unicode = event.key.keysym.unicode;
704  inst->app->getEMG()->BroadcastEventNow(kevt, inst);
705 
706  if (inst->app->getCFG()->IsDebugInfo())
707  printf("KEYBOARD: Key pressed: %d (%s)\n",
708  kevt->keysym, GetSDLName(sym));
709  inst->handle_key(event.key.keysym.scancode,1);
710  } else {
711  printf("KEYBOARD: Invalid key\n");
712  }
713  }
714  break;
715  case SDL_KEYUP :
716  {
717  SDLKey sym = event.key.keysym.sym;
718  if (sym >= SDLK_FIRST && sym < SDLK_LAST) {
719  // Mark the key as unheld
720  inst->keyheld[sym] = 0;
721 
722  // Now generate an input event..
723  KeyInputEvent *kevt = (KeyInputEvent *)
724  Event::GetEventByType(T_EV_Input_Key);
725  kevt->down = 0;
726  kevt->keysym = sym;
727  kevt->unicode = event.key.keysym.unicode;
728  inst->app->getEMG()->BroadcastEventNow(kevt, inst);
729 
730  if (inst->app->getCFG()->IsDebugInfo())
731  printf("KEYBOARD: Key released: %d (%s)\n",
732  kevt->keysym, GetSDLName(sym));
733  inst->handle_key(event.key.keysym.scancode,0);
734  break;
735  } else
736  printf("KEYBOARD: Invalid key\n");
737  }
738  break;
739  }
740  }
741  /*else {
742  printf("SDL Error Waiting for Event!\n");
743  sdlthreadgo = 0;
744  }*/
745 
746  // usleep(100); // Give up the processor briefly!
747  }
748 
749  inst->app->getEMG()->UnlistenEvent(inst,0,T_EV_ExitSession);
750 
751  printf("Closing joysticks.\n");
752  for (int i = 0; i < inst->numjoy; i++)
753  SDL_JoystickClose(inst->joys[i]);
754 
755 #ifdef __MACOSX__
756  // inst->cocoa.TakedownCocoaThread();
757 #endif
758 
759  printf("SDLIO: SDL Input thread done.\n");
760  return 0;
761 }
