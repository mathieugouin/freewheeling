80  ceil((val) * ((max) - (min)) * 0.01 + (min))

82  (NO_CHECK ? (val) : ((val < min) ? (min) : (val > max) ? (max) : (val)))

84 long HardwareMixerInterface::get_integer(char **ptr, long min, long max)
85 {
86  long val = min;
87  char *p = *ptr, *s;
88 
89  if (*p == ':')
90  p++;
91  if (*p == '\0' || (!isdigit(*p) && *p != '-'))
92  goto out;
93 
94  s = p;
95  val = strtol(s, &p, 10);
96  if (*p == '.') {
97  p++;
98  strtol(p, &p, 10);
99  }
100  if (*p == '%') {
101  val = (long)convert_prange1(strtod(s, NULL), min, max);
102  p++;
103  }
104  val = check_range(val, min, max);
105  if (*p == ',')
106  p++;
107  out:
108  *ptr = p;
109  return val;
110 }

140 int HardwareMixerInterface::parse_control_id(const char *str, snd_ctl_elem_id_t *id)
141 {
142  int c, size, numid;
143  char *ptr;
144 
145  while (*str == ' ' || *str == '\t')
146  str++;
147  if (!(*str))
148  return -EINVAL;
149  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_MIXER); /* default */
150  while (*str) {
151  if (!strncasecmp(str, "numid=", 6)) {
152  str += 6;
153  numid = atoi(str);
154  if (numid <= 0) {
155  printf( "amixer: Invalid numid %d\n", numid);
156  return -EINVAL;
157  }
158  snd_ctl_elem_id_set_numid(id, atoi(str));
159  while (isdigit(*str))
160  str++;
161  } else if (!strncasecmp(str, "iface=", 6)) {
162  str += 6;
163  if (!strncasecmp(str, "card", 4)) {
164  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_CARD);
165  str += 4;
166  } else if (!strncasecmp(str, "mixer", 5)) {
167  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_MIXER);
168  str += 5;
169  } else if (!strncasecmp(str, "pcm", 3)) {
170  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_PCM);
171  str += 3;
172  } else if (!strncasecmp(str, "rawmidi", 7)) {
173  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_RAWMIDI);
174  str += 7;
175  } else if (!strncasecmp(str, "timer", 5)) {
176  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_TIMER);
177  str += 5;
178  } else if (!strncasecmp(str, "sequencer", 9)) {
179  snd_ctl_elem_id_set_interface(id, SND_CTL_ELEM_IFACE_SEQUENCER);
180  str += 9;
181  } else {
182  return -EINVAL;
183  }
184  } else if (!strncasecmp(str, "name=", 5)) {
185  char buf[64];
186  str += 5;
187  ptr = buf;
188  size = 0;
189  if (*str == '\'' || *str == '\"') {
190  c = *str++;
191  while (*str && *str != c) {
192  if (size < (int)sizeof(buf)) {
193  *ptr++ = *str;
194  size++;
195  }
196  str++;
197  }
198  if (*str == c)
199  str++;
200  } else {
201  while (*str && *str != ',') {
202  if (size < (int)sizeof(buf)) {
203  *ptr++ = *str;
204  size++;
205  }
206  str++;
207  }
208  *ptr = '\0';
209  }
210  snd_ctl_elem_id_set_name(id, buf);
211  } else if (!strncasecmp(str, "index=", 6)) {
212  str += 6;
213  snd_ctl_elem_id_set_index(id, atoi(str));
214  while (isdigit(*str))
215  str++;
216  } else if (!strncasecmp(str, "device=", 7)) {
217  str += 7;
218  snd_ctl_elem_id_set_device(id, atoi(str));
219  while (isdigit(*str))
220  str++;
221  } else if (!strncasecmp(str, "subdevice=", 10)) {
222  str += 10;
223  snd_ctl_elem_id_set_subdevice(id, atoi(str));
224  while (isdigit(*str))
225  str++;
226  }
227  if (*str == ',') {
228  str++;
229  } else {
230  if (*str)
231  return -EINVAL;
232  }
233  }
234  return 0;
235 }

271 void HardwareMixerInterface::decode_tlv(unsigned int spaces, unsigned int *tlv, unsigned int tlv_size)
272 {
273  unsigned int type = tlv[0];
274  unsigned int size;
275  unsigned int idx = 0;
276 
277  if (tlv_size < 2 * sizeof(unsigned int)) {
278  printf("TLV size error!\n");
279  return;
280  }
281  print_spaces(spaces);
282  printf("| ");
283  type = tlv[idx++];
284  size = tlv[idx++];
285  tlv_size -= 2 * sizeof(unsigned int);
286  if (size > tlv_size) {
287  printf("TLV size error (%i, %i, %i)!\n", type, size, tlv_size);
288  return;
289  }
290  switch (type) {
291  case SND_CTL_TLVT_CONTAINER:
292  size += sizeof(unsigned int) -1;
293  size /= sizeof(unsigned int);
294  while (idx < size) {
295  if (tlv[idx+1] > (size - idx) * sizeof(unsigned int)) {
296  printf("TLV size error in compound!\n");
297  return;
298  }
299  decode_tlv(spaces + 2, tlv + idx, tlv[idx+1]);
300  idx += 2 + (tlv[1] + sizeof(unsigned int) - 1) / sizeof(unsigned int);
301  }
302  break;
303  case SND_CTL_TLVT_DB_SCALE:
304  printf("dBscale-");
305  if (size != 2 * sizeof(unsigned int)) {
306  while (size > 0) {
307  printf("0x%08x,", tlv[idx++]);
308  size -= sizeof(unsigned int);
309  }
310  } else {
311  printf("min=");
312  print_dB((int)tlv[2]);
313  printf(",step=");
314  print_dB(tlv[3] & 0xffff);
315  printf(",mute=%i", (tlv[3] >> 16) & 1);
316  }
317  break;
318 #ifdef SND_CTL_TLVT_DB_LINEAR
319  case SND_CTL_TLVT_DB_LINEAR:
320  printf("dBlinear-");
321  if (size != 2 * sizeof(unsigned int)) {
322  while (size > 0) {
323  printf("0x%08x,", tlv[idx++]);
324  size -= sizeof(unsigned int);
325  }
326  } else {
327  printf("min=");
328  print_dB(tlv[2]);
329  printf(",max=");
330  print_dB(tlv[3]);
331  }
332  break;
333 #endif
334 #ifdef SND_CTL_TLVT_DB_RANGE
335  case SND_CTL_TLVT_DB_RANGE:
336  printf("dBrange-\n");
337  if ((size / (6 * sizeof(unsigned int))) != 0) {
338  while (size > 0) {
339  printf("0x%08x,", tlv[idx++]);
340  size -= sizeof(unsigned int);
341  }
342  break;
343  }
344  idx = 0;
345  while (idx < size) {
346  print_spaces(spaces + 2);
347  printf("rangemin=%i,", tlv[0]);
348  printf(",rangemax=%i\n", tlv[1]);
349  decode_tlv(spaces + 4, tlv + 2, 6 * sizeof(unsigned int));
350  idx += 6 * sizeof(unsigned int);
351  }
352  break;
353 #endif
354 #ifdef SND_CTL_TLVT_DB_MINMAX
355  case SND_CTL_TLVT_DB_MINMAX:
356  case SND_CTL_TLVT_DB_MINMAX_MUTE:
357  if (type == SND_CTL_TLVT_DB_MINMAX_MUTE)
358  printf("dBminmaxmute-");
359  else
360  printf("dBminmax-");
361  if (size != 2 * sizeof(unsigned int)) {
362  while (size > 0) {
363  printf("0x%08x,", tlv[idx++]);
364  size -= sizeof(unsigned int);
365  }
366  } else {
367  printf("min=");
368  print_dB(tlv[2]);
369  printf(",max=");
370  print_dB(tlv[3]);
371  }
372  break;
373 #endif
374  default:
375  printf("unk-%i-", type);
376  while (size > 0) {
377  printf("0x%08x,", tlv[idx++]);
378  size -= sizeof(unsigned int);
379  }
380  break;
381  }
382  putc('\n', stdout);
383 }

385 int HardwareMixerInterface::show_control(char *card, const char *space, snd_hctl_elem_t *elem,
386  int level)
387 {
388  int err;
389  unsigned int item, idx, count, *tlv;
390  snd_ctl_elem_type_t type;
391  snd_ctl_elem_id_t *id;
392  snd_ctl_elem_info_t *info;
393  snd_ctl_elem_value_t *control;
394  snd_aes_iec958_t iec958;
395  snd_ctl_elem_id_alloca(&id);
396  snd_ctl_elem_info_alloca(&info);
397  snd_ctl_elem_value_alloca(&control);
398  if ((err = snd_hctl_elem_info(elem, info)) < 0) {
399  printf("Control %s snd_hctl_elem_info error: %s\n", card, snd_strerror(err));
400  return err;
401  }
402  if (level & LEVEL_ID) {
403  snd_hctl_elem_get_id(elem, id);
404  show_control_id(id);
405  printf("\n");
406  }
407  count = snd_ctl_elem_info_get_count(info);
408  type = snd_ctl_elem_info_get_type(info);
409  printf("%s; type=%s,access=%s,values=%i", space, control_type(info), control_access(info), count);
410  switch (type) {
411  case SND_CTL_ELEM_TYPE_INTEGER:
412  printf(",min=%li,max=%li,step=%li\n",
413  snd_ctl_elem_info_get_min(info),
414  snd_ctl_elem_info_get_max(info),
415  snd_ctl_elem_info_get_step(info));
416  break;
417  case SND_CTL_ELEM_TYPE_INTEGER64:
418  printf(",min=%Li,max=%Li,step=%Li\n",
419  snd_ctl_elem_info_get_min64(info),
420  snd_ctl_elem_info_get_max64(info),
421  snd_ctl_elem_info_get_step64(info));
422  break;
423  case SND_CTL_ELEM_TYPE_ENUMERATED:
424  {
425  unsigned int items = snd_ctl_elem_info_get_items(info);
426  printf(",items=%u\n", items);
427  for (item = 0; item < items; item++) {
428  snd_ctl_elem_info_set_item(info, item);
429  if ((err = snd_hctl_elem_info(elem, info)) < 0) {
430  printf("Control %s element info error: %s\n", card, snd_strerror(err));
431  return err;
432  }
433  printf("%s; Item #%u '%s'\n", space, item, snd_ctl_elem_info_get_item_name(info));
434  }
435  break;
436  }
437  default:
438  printf("\n");
439  break;
440  }
441  if (level & LEVEL_BASIC) {
442  if (!snd_ctl_elem_info_is_readable(info))
443  goto __skip_read;
444  if ((err = snd_hctl_elem_read(elem, control)) < 0) {
445  printf("Control %s element read error: %s\n", card, snd_strerror(err));
446  return err;
447  }
448  printf("%s: values=", space);
449  for (idx = 0; idx < count; idx++) {
450  if (idx > 0)
451  printf(",");
452  switch (type) {
453  case SND_CTL_ELEM_TYPE_BOOLEAN:
454  printf("%s", snd_ctl_elem_value_get_boolean(control, idx) ? "on" : "off");
455  break;
456  case SND_CTL_ELEM_TYPE_INTEGER:
457  printf("%li", snd_ctl_elem_value_get_integer(control, idx));
458  break;
459  case SND_CTL_ELEM_TYPE_INTEGER64:
460  printf("%Li", snd_ctl_elem_value_get_integer64(control, idx));
461  break;
462  case SND_CTL_ELEM_TYPE_ENUMERATED:
463  printf("%u", snd_ctl_elem_value_get_enumerated(control, idx));
464  break;
465  case SND_CTL_ELEM_TYPE_BYTES:
466  printf("0x%02x", snd_ctl_elem_value_get_byte(control, idx));
467  break;
468  case SND_CTL_ELEM_TYPE_IEC958:
469  snd_ctl_elem_value_get_iec958(control, &iec958);
470  printf("[AES0=0x%02x AES1=0x%02x AES2=0x%02x AES3=0x%02x]",
471  iec958.status[0], iec958.status[1],
472  iec958.status[2], iec958.status[3]);
473  break;
474  default:
475  printf("?");
476  break;
477  }
478  }
479  printf("\n");
480  __skip_read:
481  if (!snd_ctl_elem_info_is_tlv_readable(info))
482  goto __skip_tlv;
483  tlv = (unsigned int *) malloc(4096);
484  if ((err = snd_hctl_elem_tlv_read(elem, tlv, 4096)) < 0) {
485  printf("Control %s element TLV read error: %s\n", card, snd_strerror(err));
486  free(tlv);
487  return err;
488  }
489  decode_tlv(strlen(space), tlv, 4096);
490  free(tlv);
491  }
492  __skip_tlv:
493  return 0;
494 }

524 int HardwareMixerInterface::cset(char *card, int argc, char *argv[], int roflag, int keep_handle, int debugflag)
525 {
526  int err;
527  static snd_ctl_t *handle = NULL;
528  snd_ctl_elem_info_t *info;
529  snd_ctl_elem_id_t *id;
530  snd_ctl_elem_value_t *control;
531  char *ptr;
532  unsigned int idx, count;
533  long tmp;
534  snd_ctl_elem_type_t type;
535  snd_ctl_elem_info_alloca(&info);
536  snd_ctl_elem_id_alloca(&id);
537  snd_ctl_elem_value_alloca(&control);
538 
539  if (argc < 1) {
540  printf( "Specify a full control identifier: [[iface=<iface>,][name='name',][index=<index>,][device=<device>,][subdevice=<subdevice>]]|[numid=<numid>]\n");
541  return -EINVAL;
542  }
543  if (parse_control_id(argv[0], id)) {
544  printf( "Wrong control identifier: %s\n", argv[0]);
545  return -EINVAL;
546  }
547  if (debugflag) {
548  printf("VERIFY ID: ");
549  show_control_id(id);
550  printf("\n");
551  }
552  if (handle == NULL &&
553  (err = snd_ctl_open(&handle, card, 0)) < 0) {
554  printf("Control %s open error: %s\n", card, snd_strerror(err));
555  return err;
556  }
557  snd_ctl_elem_info_set_id(info, id);
558  if ((err = snd_ctl_elem_info(handle, info)) < 0) {
559  printf("Cannot find the given element from control %s\n", card);
560  if (! keep_handle) {
561  snd_ctl_close(handle);
562  handle = NULL;
563  }
564  return err;
565  }
566  snd_ctl_elem_info_get_id(info, id); /* FIXME: Remove it when hctl find works ok !!! */
567  type = snd_ctl_elem_info_get_type(info);
568  count = snd_ctl_elem_info_get_count(info);
569  snd_ctl_elem_value_set_id(control, id);
570 
571  if (!roflag) {
572  ptr = argv[1];
573  for (idx = 0; idx < count && idx < 128 && ptr && *ptr; idx++) {
574  switch (type) {
575  case SND_CTL_ELEM_TYPE_BOOLEAN:
576  tmp = 0;
577  if (!strncasecmp(ptr, "on", 2) || !strncasecmp(ptr, "up", 2)) {
578  tmp = 1;
579  ptr += 2;
580  } else if (!strncasecmp(ptr, "yes", 3)) {
581  tmp = 1;
582  ptr += 3;
583  } else if (!strncasecmp(ptr, "toggle", 6)) {
584  tmp = snd_ctl_elem_value_get_boolean(control, idx);
585  tmp = tmp > 0 ? 0 : 1;
586  ptr += 6;
587  } else if (isdigit(*ptr)) {
588  tmp = atoi(ptr) > 0 ? 1 : 0;
589  while (isdigit(*ptr))
590  ptr++;
591  } else {
592  while (*ptr && *ptr != ',')
593  ptr++;
594  }
595  snd_ctl_elem_value_set_boolean(control, idx, tmp);
596  break;
597  case SND_CTL_ELEM_TYPE_INTEGER:
598  tmp = get_integer(&ptr,
599  snd_ctl_elem_info_get_min(info),
600  snd_ctl_elem_info_get_max(info));
601  snd_ctl_elem_value_set_integer(control, idx, tmp);
602  break;
603  case SND_CTL_ELEM_TYPE_INTEGER64:
604  tmp = get_integer64(&ptr,
605  snd_ctl_elem_info_get_min64(info),
606  snd_ctl_elem_info_get_max64(info));
607  snd_ctl_elem_value_set_integer64(control, idx, tmp);
608  break;
609  case SND_CTL_ELEM_TYPE_ENUMERATED:
610  tmp = get_ctl_enum_item_index(handle, info, &ptr);
611  if (tmp < 0)
612  tmp = get_integer(&ptr, 0, snd_ctl_elem_info_get_items(info) - 1);
613  snd_ctl_elem_value_set_enumerated(control, idx, tmp);
614  break;
615  case SND_CTL_ELEM_TYPE_BYTES:
616  tmp = get_integer(&ptr, 0, 255);
617  snd_ctl_elem_value_set_byte(control, idx, tmp);
618  break;
