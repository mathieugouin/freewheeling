650 // Ringbuffer implementation for a Single Reader Thread & Multiple Writer Threads
651 //
652 // This expands to a set of ringbuffers- one for each writer thread. The order of elements between threads is not preserved.
653 // Elements are written and read using COPY operations.
654 //
655 // NOTE: Writer threads should not be added and removed once multithreaded operation has begun.
656 // New writer threads may be registered after the ringbuffer is created, but only while that RingBuffer
657 // is being written to by a single thread (ie during a single initialization thread).
658 //
659 // Class T must be able to be assigned to a null integer.
660 // If class T is a pointer, no problem.
661 // If class T is an instance, you must provide an initializer accepting an integer.
662 
663 template <class T> class SRMWRingBuffer : public RTDataStruct_Updater {
664  friend class RT_RWThreads;
665 
666 public:
667 
668  // Create a ringbuffer with numel elements of class T
669  SRMWRingBuffer (int numel) : numel(numel), num_writers(RT_RWThreads::num_rw_threads) {
670  wbufs = new jack_ringbuffer_t *[MAX_RW_THREADS];
671  for (int i = 0; i < num_writers; i++)
672  wbufs[i] = jack_ringbuffer_create(sizeof(T) * numel);
673 
674  // Register this ringbuf
675  RT_RWThreads::RegisterRTDataStruct(this);
676  };
677  virtual ~SRMWRingBuffer() {
678  // Unregister this ringbuf
679  RT_RWThreads::UnregisterRTDataStruct(this);
680 
681  for (int i = 0; i < num_writers; i++)
682  jack_ringbuffer_free(wbufs[i]);
683  delete[] wbufs;
684  };

687 
688  int WriteElement (const T &el) {
689  if (num_writers != RT_RWThreads::num_rw_threads) {
690  pthread_mutex_lock(&RT_RWThreads::register_rtstruct_lock);
691  pthread_mutex_unlock(&RT_RWThreads::register_rtstruct_lock);
692  if (num_writers != RT_RWThreads::num_rw_threads) {
693  printf("CORE: ERROR: SRMWRingBuffer thread count mismatch.\n");
694  exit(1);
695  }
696  }
697 
698  // Determine which write thread we are
699  pthread_t id = pthread_self();
700  for (int i = 0; i < num_writers; i++)
701  if (pthread_equal(id,RT_RWThreads::ids[i])) {
702  // Write to the appropriate ringbuf
703  if (jack_ringbuffer_write(wbufs[i],(const char *) &el,sizeof(T)) < sizeof(T)) {
704  printf("CORE: No space in RingBuffer for element\n");
705  return -1;
706  } else
707  return 0;
708  }
709

711  return -1;

765 // An RTStore is the concept of a store shelf stocked with cans of something (class T)
766 // The store shelf has a number of items. Each item has a state. The state can be compared and swapped
767 // atomically. This allows several threads to work together without locks, stocking and pulling items
768 // from the shelf.
769 //
770 // RTStore does not care about the content of class T. It does not care about the cans.
771 // It only cares about managing the state of each item on the shelf in a thread-safe, RT-safe way.
772 //
773 // This design pattern is used in the memory manager classes (fweelin_mem).
774 template <class T> class RTStore {
775 public:
776  // 3 states of the transaction at a given index- waiting, busy and done.
777  const static int ITEM_WAITING = 0,
778  ITEM_BUSY = 1,
779  ITEM_DONE = 2;
780 
781 private:
782  class RTStoreItem : public T {
783  public:
784  RTStoreItem() : item_status(ITEM_DONE) {};
785 
786  volatile int item_status; // Current status of this item (update using careful atomic ops)
787  };
788 
789 public:
790  RTStore (int num_items) : num_items(num_items) {
791  items = new RTStoreItem[num_items];
792  };
793  ~RTStore () {
794  delete [] items;
795  };
796 
797  // Finds the first RTStoreItem with the given state 'find_state' and resets the state atomically to
798  // 'replace_state'. Returns the index of the store item in the store (in idx), as well as a pointer to the
799  // item object itself (return).
800  //
801  // Returns null if no item could be found with the given state.
802  inline T *FindItemWithState (int find_state, int replace_state, int &idx) {
803  for (int i = 0; i < num_items; i++) {
804  // printf("RTStore Find (%p): %d: state is %d : find %d replace %d\n",this,i,items[i].item_status,find_state,replace_state);
805 
806  // Compare and swap, atomically. This guarantees that only one thread can modify an RTStoreItem
807  // at a time, without locking.
808  if (__sync_bool_compare_and_swap(&items[i].item_status,find_state,replace_state)) {
809  // printf(" got it\n");
810 
811  // Found.
812  idx = i;
813  return &items[i];
814  }
815  }
816 
817  // None found
818  return 0;
819  };
820 
821  // Change the state of store item with given index to 'new_state', if the item has the expected state.
822  // Returns nonzero if the state matched and was changed. Returns zero if the state did not match.
823  inline char ChangeStateAtIdx (int idx, int expect_state, int new_state) {
824  // printf("RTStore Change (%p): %d: state is %d : expect %d replace %d\n",this,idx,items[idx].item_status,expect_state,new_state);
825  return __sync_bool_compare_and_swap(&items[idx].item_status,expect_state,new_state);
826  };
827 
828  // Returns the item at the given index. To modify the item, change to the BUSY state first.
829  inline T *GetItemAtIdx (int idx) { return &items[idx]; };
830 
831 private:
832 
833  RTStoreItem *items;
834  int num_items; // Number of items in the store
835 };
