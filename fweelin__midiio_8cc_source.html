207 void MidiIO::MidiInputProc (const MIDIPacketList *pktlist, void *refCon, void *connRefCon) {
208  MidiIO *inst = static_cast<MidiIO *>(refCon);
209 
210  // Check whether to unbypass used channels
211  inst->CheckUnbypass();
212 
213  MIDIPacket *packet = (MIDIPacket *) pktlist->packet;
214  for (int j = 0; j < pktlist->numPackets; j++) {
215  if ((packet->data[0] >= 0x80 && packet->data[0] <= 0x8F) ||
216  (packet->data[0] >= 0x90 && packet->data[0] <= 0x9F &&
217  packet->data[2] == 0)) { // Interpret NoteOn velocity 0 as note off
218  // Note Off
219  int base = (packet->data[0] >= 0x90 ? 0x90 : 0x80);
220  inst->ReceiveNoteOffEvent(packet->data[0] - base,
221  packet->data[1],
222  packet->data[2]);
223  } else if (packet->data[0] >= 0x90 && packet->data[0] <= 0x9F) {
224  // Note On
225  inst->ReceiveNoteOnEvent(packet->data[0] - 0x90,
226  packet->data[1],
227  packet->data[2]);
228  } else if (packet->data[0] >= 0xB0 && packet->data[0] <= 0xBF) {
229  // Control Change
230  inst->ReceiveControlChangeEvent(packet->data[0] - 0xB0,
231  packet->data[1],
232  packet->data[2]);
233  } else if (packet->data[0] >= 0xC0 && packet->data[0] <= 0xCF) {
234  // Program Change
235  inst->ReceiveProgramChangeEvent(packet->data[0] - 0xC0,
236  packet->data[1]);
237  } else if (packet->data[0] >= 0xD0 && packet->data[0] <= 0xDF) {
238  // Channel Pressure
239  inst->ReceiveChannelPressureEvent(packet->data[0] - 0xD0,
240  packet->data[1]);
241  } else if (packet->data[0] >= 0xE0 && packet->data[0] <= 0xEF) {
242  // Pitch Bend
243  int lsb = packet->data[1],
244  msb = packet->data[2],
245  benderval = msb << 8 + lsb;
246 
247  inst->ReceivePitchBendEvent(packet->data[0] - 0xE0,
248  benderval);
249  }
250 
251  // Advance
252  packet = MIDIPacketNext(packet);
253  }
254 
255  // Check whether to bypass unused channels
256  inst->CheckBypass();
257 }

458 void *MidiIO::run_midi_thread(void *ptr)
459 {
460  MidiIO *inst = static_cast<MidiIO *>(ptr);
461  int npfd;
462  struct pollfd *pfd;
463 
464  FloConfig *fs = inst->app->getCFG();
465  inst->checkfreq = inst->app->getAUDIO()->get_srate(); // How often to check auto-bypass conditions for MIDI
466 
467  printf("MIDIthread start..\n");
468  // printf("*** MIDI THREAD: %li\n",pthread_self());
469 
470  // Open up ALSA MIDI client
471  if (inst->open_midi(1, fs->GetNumMIDIOuts())) {
472  fprintf(stderr, "MIDI: Can't open ALSA MIDI.\n");
473  exit(1);
474  }
475  npfd = snd_seq_poll_descriptors_count(inst->seq_handle, POLLIN);
476  pfd = (struct pollfd *)alloca(npfd * sizeof(struct pollfd));
477  snd_seq_poll_descriptors(inst->seq_handle, pfd, npfd, POLLIN);
478 
479  inst->listen_events();
480 
481  // Main MIDI loop
482  while (inst->midithreadgo) {
483  // Every second, come out of poll-- so we don't get locked up
484  // if no MIDI events come in
485  if (poll(pfd, npfd, 1000) > 0) {
486  // There's an event here!
487 
488  // Check whether to unbypass used channels
489  inst->CheckUnbypass();
490 
491  // Read event(s)
492  snd_seq_event_t *ev;
493  // static int cnt = 0;
494  do {
495  snd_seq_event_input(inst->seq_handle, &ev);
496  switch (ev->type) {
497 #ifdef FWEELIN_LOG_UNHANDLED_MIDI_EVTS
498  case SND_SEQ_EVENT_QFRAME:
499  printf("MTC quarterframe\n");
500  break;
501 
502  case SND_SEQ_EVENT_CLOCK:
503  if (cnt % 24 == 0)
504  printf("clock: %d\n",cnt);
505  cnt++;
506  //printf("clock: %d\n",ev->data.time.tick);
507  //printf("clock: %d\n",ev->data.time.time.tv_sec);
508  //printf("clock: %d\n",ev->data.queue.param.time.time.tv_sec);
509  break;
510 
511  case SND_SEQ_EVENT_TICK:
512  printf("MIDI: 'tick' not yet implemented\n");
513  break;
514 
515  case SND_SEQ_EVENT_TEMPO:
516  printf("MIDI: 'tempo' not yet implemented\n");
517  break;
518 #endif // FWEELIN_LOG_UNHANDLED_MIDI_EVTS
519 
520  case SND_SEQ_EVENT_CONTROLLER:
521  // Control Change
522  inst->ReceiveControlChangeEvent(ev->data.control.channel,
523  ev->data.control.param,
524  ev->data.control.value);
525  break;
526 
527  case SND_SEQ_EVENT_CHANPRESS:
528  // Channel aftertouch
529  inst->ReceiveChannelPressureEvent(ev->data.control.channel,
530  ev->data.control.value);
531  break;
532 
533  case SND_SEQ_EVENT_PGMCHANGE:
534  // Program Change
535  inst->ReceiveProgramChangeEvent(ev->data.control.channel,
536  ev->data.control.value);
537  break;
538 
539  case SND_SEQ_EVENT_PITCHBEND:
540  // Pitch Bend
541  inst->ReceivePitchBendEvent(ev->data.control.channel,
542  ev->data.control.value);
543  break;
544 
545  case SND_SEQ_EVENT_NOTEON:
546  // Note On
547  inst->ReceiveNoteOnEvent(ev->data.control.channel,
548  ev->data.note.note,
549  ev->data.note.velocity);
550  break;
551 
552  case SND_SEQ_EVENT_NOTEOFF:
553  // Note Off
554  inst->ReceiveNoteOffEvent(ev->data.control.channel,
555  ev->data.note.note,
556  ev->data.note.velocity);
557  break;
558  }
559 
560  snd_seq_free_event(ev);
561  } while (snd_seq_event_input_pending(inst->seq_handle, 0) > 0);
562  }
563 
564  // Every MIDI msg or at max 1 s intervals, check auto-bypass conditions
565  inst->CheckBypass();
566  }
567 
568  inst->unlisten_events();
569 
570  printf("MIDI: thread done\n");
571  return 0;
572 }

1025 int MidiIO::EchoEventToPortChannel (Event *ev, int port, int channel, int bypasschannel) {
1026  FloConfig *fs = app->getCFG();
1027  static int midi_clock_count = 0;
1028 
1029  // DEBUG
1030  // printf("ECHO EVENT -> Port: %d Channel: %d\n",port,channel);
1031 
1032  int ret = -1;
1033  switch (ev->GetType()) {
1034  case T_EV_Input_MIDIClock :
1035  {
1036  // MIDIClockInputEvent *clkevt = (MIDIClockInputEvent *) ev;
1037  OutputClock(ret = port);
1038 
1039  midi_clock_count++;
1040  if (midi_clock_count >= MIDI_CLOCK_FREQUENCY) {
1041  midi_clock_count = 0;
1042  if (CRITTERS)
1043  printf("MIDI: Quarter Note Clock\n");
1044  }
1045  }
1046  break;
1047 
1048  case T_EV_Input_MIDIStartStop :
1049  {
1050  MIDIStartStopInputEvent *ssevt = (MIDIStartStopInputEvent *) ev;
1051  if (ssevt->start) {
1052  midi_clock_count = 0;
1053  OutputSPP(ret = port); // Output song position pointer first
1054  OutputStart(port); // Then start message
1055  } else
1056  OutputStop(ret = port);
1057  }
1058  break;
1059 
1060  case T_EV_Input_MIDIController :
1061  {
1062  MIDIControllerInputEvent *mcev = (MIDIControllerInputEvent *) ev;
1063  int p = (port == -1 ? mcev->outport-1 : port),
1064  c = (channel == -1 ? mcev->channel : channel),
1065  bc = (bypasschannel == -1 ? c : bypasschannel);
1066  ret = p;
1067 
1068  OutputController(p,c,
1069  mcev->ctrl,
1070  mcev->val);
1071 
1072  if (mcev->ctrl == MIDI_CC_SUSTAIN) {
1073  BypassInfo *b = getBP(p,bc);
1074  if (b != 0) {
1075  // printf ("sustain: %d\n",mcev->val);
1076  b->sp = mcev->val;
1077  if (b->sp == 0 && b->numheld == 0)
1078  b->releasecnt = app->getRP()->GetSampleCnt(); // Register release time of notes as when sustain pedal is released
1079  } else
1080  printf("MIDI: Can't find BypassInfo struct for p/c [%d/%d]\n",p,c);
1081  }
1082  }
1083  break;
1084 
1085  case T_EV_Input_MIDIProgramChange :
1086  {
1087  MIDIProgramChangeInputEvent *mpev = (MIDIProgramChangeInputEvent *) ev;
1088  OutputProgramChange(ret = (port == -1 ? mpev->outport-1 : port),
1089  (channel == -1 ? mpev->channel : channel),
1090  mpev->val);
1091  }
1092  break;
1093 
1094  case T_EV_Input_MIDIChannelPressure :
1095  {
1096  MIDIChannelPressureInputEvent *mpev =
1097  (MIDIChannelPressureInputEvent *) ev;
1098  OutputChannelPressure(ret = (port == -1 ? mpev->outport-1 : port),
1099  (channel == -1 ? mpev->channel : channel),
1100  mpev->val);
1101  }
1102  break;
1103 
1104  case T_EV_Input_MIDIPitchBend :
1105  {
1106  MIDIPitchBendInputEvent *mpev = (MIDIPitchBendInputEvent *) ev;
1107  OutputPitchBend(ret = (port == -1 ? mpev->outport-1 : port),
1108  (channel == -1 ? mpev->channel : channel),
1109  mpev->val);
1110  }
1111  break;
1112 
1113  case T_EV_Input_MIDIKey :
1114  {
1115  MIDIKeyInputEvent *mkev = (MIDIKeyInputEvent *) ev;
1116  int p = (port == -1 ? mkev->outport-1 : port),
1117  c = (channel == -1 ? mkev->channel : channel),
1118  bc = (bypasschannel == -1 ? c : bypasschannel);
1119  ret = p;
1120 
1121  OutputNote(p,c,
1122  mkev->down,
1123  mkev->notenum + fs->transpose,
1124  mkev->vel);
1125 
1126  BypassInfo *b = getBP(p,bc);
1127  if (b != 0) {
1128  // Keep track of # of keys held down- assume no duplicates or repeated note offs
1129  if (mkev->down) {
1130  b->notepresscnt = app->getRP()->GetSampleCnt();
1131  b->numheld++;
1132  } else {
1133  b->numheld--;
1134  if (b->numheld <= 0) {
1135  b->numheld = 0;
1136  if (b->sp == 0) // Only register release time if sustain pedal is released
1137  b->releasecnt = app->getRP()->GetSampleCnt();
1138  }
1139  }
1140 
1141  // printf ("held: %d\n",b->numheld);
1142  } else
1143  printf("MIDI: Can't find BypassInfo struct for p/c [%d/%d]\n",p,c);
1144 
1145 #ifdef FWEELIN_EXPERIMENTAL_SOFTSYNTHS_KLUDGE
1146  // For poorly behaved softsynths, double up on Note Offs
1147  if (!mkev->down)
1148  OutputNote(ret = (port == -1 ? mkev->outport-1 : port),
1149  (channel == -1 ? mkev->channel : channel),
1150  mkev->down,
1151  mkev->notenum + fs->transpose,
1152  mkev->vel);
1153 #endif // FWEELIN_EXPERIMENTAL_SOFTSYNTHS_KLUDGE
1154  }
1155  break;
1156 
1157  default:
1158  break;
1159  }
1160 
1161  return ret;

1164 void MidiIO::EchoEvent (Event *ev) {
1165  // Handle MIDI event echos..
1166  if (ev->GetType() == T_EV_Input_MIDIClock ||
1167  ev->GetType() == T_EV_Input_MIDIStartStop) {
1168  // MIDI sync message- send out to all ports set to transmit MIDI sync
1169  int *xmitports = app->getCFG()->GetMIDISyncOuts(),
1170  numxmitports = app->getCFG()->GetNumMIDISyncOuts();
1171  for (int i = 0; i < numxmitports; i++) {
1172  OutputStartOnPort();
1173  int port = EchoEventToPortChannel(ev,xmitports[i],-1,-1);
1174  OutputEndOnPort(port);
1175  }
1176  } else if (!ev->echo) {
1177  // This event is generated from the configuration explicitly- not
1178  // an automatic echo of an unhandled MIDI event--
1179  // So don't send it through our current patch settings.
1180  // Send to the exact port and channel specified
1181 
1182  OutputStartOnPort();
1183  int port = EchoEventToPortChannel(ev,-1,-1,-1);
1184  OutputEndOnPort(port);
1185  } else {
1186  // Echo of unused MIDI event- use patch logic
1187 
1188  // Handle special note off case- ensure note off sent to right place(s)
1189  PatchItem *ev_patch = 0;
1190  int ev_port = 0;
1191  if (ev->GetType() == T_EV_Input_MIDIKey &&
