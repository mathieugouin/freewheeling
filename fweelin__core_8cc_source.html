101 // Trigger snapshot #idx - return nonzero on failure
102 char Fweelin::TriggerSnapshot (int idx) {
103  Snapshot *s = getSNAP(idx);
104  if (s != 0 && s->exists) {
105  for (int i = 0; i < s->numls; i++) {
106  LoopSnapshot *ls = &(s->ls[i]);
107 
108  loopmgr->SetLoopVolume(ls->l_idx,ls->l_vol);
109 
110  if (ls->status == T_LS_Off && loopmgr->IsActive(ls->l_idx)) {
111  loopmgr->Deactivate(ls->l_idx);
112  } else if (ls->status == T_LS_Playing ||
113  ls->status == T_LS_Overdubbing) {
114  if (loopmgr->GetStatus(ls->l_idx) != T_LS_Playing) {
115  // Loop not yet playing
116  if (loopmgr->IsActive(ls->l_idx))
117  loopmgr->Deactivate(ls->l_idx);
118  loopmgr->Activate(ls->l_idx, 0, ls->t_vol);
119  } else {
120  // Loop already playing- adjust volume
121  loopmgr->SetTriggerVol(ls->l_idx, ls->t_vol);
122  }
123  }
124  }
125 
126  return 0;
127  } else
128  return 1;

131 // Splits a saveable filename in the format 'basename-hash-objectname'
132 // into its base name, hash and object name components
133 //
134 // Returns zero on success
135 char Saveable::SplitFilename(const char *filename, int baselen, char *basename,
136  char *hash, char *objname,
137  int maxlen) {
138  // Loop exists, use combination of time and hash as name
139  const char *slashptr = filename + baselen;
140 
141  if (slashptr < filename+strlen(filename)) {
142  const char *slashptr2 = strchr(slashptr+1,'-'),
143  *extptr = strrchr(filename,'.');
144 
145  if (extptr == 0)
146  extptr = filename+strlen(filename); // No extension
147 
148  // Extract base name
149  int len = 0;
150  if (basename != 0) {
151  len = MIN(baselen,maxlen-1);
152  memcpy(basename,filename,sizeof(char)*len);
153  basename[len] = '\0';
154  }
155 
156  // Extract hash
157  const char *breaker = (slashptr2 != 0 ? slashptr2 : extptr);
158  if (strlen(slashptr+1) - strlen(breaker) == SAVEABLE_HASH_LENGTH*2) {
159  if (hash != 0) {
160  len = MIN(SAVEABLE_HASH_LENGTH*2,maxlen-1);
161  memcpy(hash,slashptr+1,sizeof(char)*len);
162  hash[len] = '\0';
163  }
164  } else {
165  printf("SAVEABLE: Invalid hash within filename: '%s'\n",
166  filename);
167  return 1;
168  }
169 
170  // Now check if the filename also contains an object name-
171  // this would be placed after the hash
172  if (objname != 0) {
173  if (slashptr2 != 0) {
174  // Name
175  len = strlen(slashptr2+1) - strlen(extptr);
176  len = MIN(len,maxlen-1);
177  memcpy(objname,slashptr2+1,sizeof(char)*len);
178  objname[len] = '\0';
179  } else
180  strcpy(objname,"");
181  }
182  } else {
183  printf("SAVEABLE: Invalid filename for extracting hash/name: '%s'\n",
184  filename);
185  return 1;
186  }
187 
188  return 0;
189 };

206  delete[] *filename_ptr;

208  strlen(fn_hash)+1+
209  strlen(newname)+1];

213  else
214  sprintf(*filename_ptr,"%s-%s",
215  fn_base,fn_hash);
216 
217  unsigned int tmp_a_size = FWEELIN_OUTNAME_LEN + 10;
218  unsigned int tmp_b_size = FWEELIN_OUTNAME_LEN;
219  char tmp_a[tmp_a_size],
220  tmp_b[tmp_b_size];

222  // Add each type of extension provided and rename the file

224  snprintf(tmp_b,tmp_b_size,"%s%s",*filename_ptr,exts[i]);
225 
226  if (!rename(tmp_a,tmp_b))
227  printf("SAVEABLE: Rename file '%s' -> '%s'\n",tmp_a,tmp_b);

229  //printf("SAVEABLE: File '%s' not found for rename\n",tmp_a);

430 void LoopManager::ItemRenamed(BrowserItem *item) {
431  switch (item->GetType()) {
432  case B_Loop_Tray:
433  // Change name inside the loop
434  {
435  LoopTrayItem *curl = (LoopTrayItem *) item;
436  char *old_filename = 0,
437  *new_filename = 0;
438 
439  // Rename on disk
440  const static char *exts[] = {app->getCFG()->GetAudioFileExt(curl->l->format),
441  FWEELIN_OUTPUT_DATA_EXT};
442  curl->l->RenameSaveable(app->getCFG()->GetLibraryPath(),
443  FWEELIN_OUTPUT_LOOP_NAME,
444  curl->l->name, curl->name,
445  exts, 2,
446  &old_filename,
447  &new_filename);
448 
449  // We also need to rename in the loop browser
450  if (app->getBROWSER(B_Loop) != 0)
451  app->getBROWSER(B_Loop)->
452  ItemRenamedOnDisk(old_filename,new_filename,curl->name);
453 
454  if (old_filename != 0)
455  delete[] old_filename;
456  if (new_filename != 0)
457  delete[] new_filename;
458 
459  // And in memory..
460  RenameLoop(curl->l,curl->name);
461  }
462  break;
463 
464  case B_Loop:
465  {
466  // Name change on disk
467  int baselen = strlen(app->getCFG()->GetLibraryPath()) + 1 +
468  strlen(FWEELIN_OUTPUT_LOOP_NAME);
469 
470  // Add all audio format names + XML to extension list
471  int numexts = END_OF_FORMATS + 1;
472  char *exts[numexts];
473  for (codec c = FIRST_FORMAT; c < END_OF_FORMATS; c = (codec) (c+1))
474  exts[c] = app->getCFG()->GetAudioFileExt(c);
475  exts[END_OF_FORMATS] = FWEELIN_OUTPUT_DATA_EXT;
476 
477  // Rename all possible audio files + XML metadata to new name
478  printf("DISK: Rename '%s'\n",((LoopBrowserItem *) item)->filename);
479  Saveable::RenameSaveable(&((LoopBrowserItem *) item)->filename,baselen,
480  item->name,(const char **) exts,numexts);
