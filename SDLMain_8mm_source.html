296 /*
297  * Catch document open requests...this lets us notice files when the app
298  * was launched by double-clicking a document, or when a document was
299  * dragged/dropped on the app's icon. You need to have a
300  * CFBundleDocumentsType section in your Info.plist to get this message,
301  * apparently.
302  *
303  * Files are added to gArgv, so to the app, they'll look like command line
304  * arguments. Previously, apps launched from the finder had nothing but
305  * an argv[0].
306  *
307  * This message may be received multiple times to open several docs on launch.
308  *
309  * This message is ignored once the app's mainline has been called.
310  */
311 - (BOOL)application:(NSApplication *)theApplication openFile:(NSString *)filename
312 {
313  const char *temparg;
314  size_t arglen;
315  char *arg;
316  char **newargv;
317 
318  if (!gFinderLaunch) /* MacOS is passing command line args. */
319  return FALSE;
320 
321  if (gCalledAppMainline) /* app has started, ignore this document. */
322  return FALSE;
323 
324  temparg = [filename UTF8String];
325  arglen = SDL_strlen(temparg) + 1;
326  arg = (char *) SDL_malloc(arglen);
327  if (arg == NULL)
328  return FALSE;
329 
330  newargv = (char **) realloc(gArgv, sizeof (char *) * (gArgc + 2));
331  if (newargv == NULL)
332  {
333  SDL_free(arg);
334  return FALSE;
335  }
336  gArgv = newargv;
337 
338  SDL_strlcpy(arg, temparg, arglen);
339  gArgv[gArgc++] = arg;
340  gArgv[gArgc] = NULL;
341  return TRUE;
342 }

370 - (NSString *)stringByReplacingRange:(NSRange)aRange with:(NSString *)aString
371 {
372  unsigned int bufferSize;
373  unsigned int selfLen = [self length];
374  unsigned int aStringLen = [aString length];
375  unichar *buffer;
376  NSRange localRange;
377  NSString *result;
378 
379  bufferSize = selfLen + aStringLen - aRange.length;
380  buffer = (unichar *) NSAllocateMemoryPages(bufferSize*sizeof(unichar));
381 
382  /* Get first part into buffer */
383  localRange.location = 0;
384  localRange.length = aRange.location;
385  [self getCharacters:buffer range:localRange];
386 
387  /* Get middle part into buffer */
388  localRange.location = 0;
389  localRange.length = aStringLen;
390  [aString getCharacters:(buffer+aRange.location) range:localRange];
391 
392  /* Get last part into buffer */
393  localRange.location = aRange.location + aRange.length;
394  localRange.length = selfLen - localRange.location;
395  [self getCharacters:(buffer+aRange.location+aStringLen) range:localRange];
396 
397  /* Build output string */
398  result = [NSString stringWithCharacters:buffer length:bufferSize];
399 
400  NSDeallocateMemoryPages(buffer, bufferSize);
401 
402  return result;
403 }
