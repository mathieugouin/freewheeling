43 MemoryManager::MemoryManager() : update_queue(0) {
44  // Init mutex/conditions
45  pthread_mutex_init(&mgr_thread_lock,0);
46  pthread_cond_init(&mgr_go,0);
47 
48  // Create update queue
49  update_queue = new SRMWRingBuffer<MemoryManagerUpdate>(MEMMGR_UPDATE_QUEUE_SIZE);
50 
51  const static size_t STACKSIZE = 1024*128;
52  pthread_attr_t attr;
53  pthread_attr_init(&attr);
54  pthread_attr_setstacksize(&attr,STACKSIZE);
55  printf("MEMMGR: Stacksize: %zd.\n",STACKSIZE);
56 
57  // Setup manager thread
58  threadgo = 1;
59  int ret = pthread_create(&mgr_thread,
60  &attr,
61  run_mgr_thread,
62  static_cast<void *>(this));
63  if (ret != 0) {
64  printf("MEM: (memorymanager) pthread_create failed, exiting");
65  exit(1);
66  }
67  RT_RWThreads::RegisterReaderOrWriter(mgr_thread);
68 
69  // Setup high priority threads
70  struct sched_param schp;
71  memset(&schp, 0, sizeof(schp));
72  schp.sched_priority = sched_get_priority_max(SCHED_OTHER);
73  printf("MEM: Memory manager thread (p%d)\n",schp.sched_priority);
74  if (pthread_setschedparam(mgr_thread, SCHED_OTHER, &schp) != 0) {
75  printf("MEM: Can't set thread priority, will use regular!\n");
76  }
77 };

121 
122  MemoryManagerUpdate mmu = 0;

124  // printf("MEM: ...Processing queue...\n");

144  // printf("MEM: end process queue\n");

147 void *MemoryManager::run_mgr_thread (void *ptr) {
148  MemoryManager *inst = static_cast<MemoryManager *>(ptr);
149 
150  // printf("*** THREAD: %p\n",pthread_self());
151 
152  pthread_mutex_lock(&inst->mgr_thread_lock);
153  while (inst->threadgo) {
154  inst->ProcessQueue();
155 
156  // Wait for wakeup
157  pthread_cond_wait (&inst->mgr_go, &inst->mgr_thread_lock);
158 
159  inst->needs_wakeup = 0; // Woken!
160  }
161 
162  printf("MEM: Begin cleanup.\n");
163  inst->ProcessQueue();
164 
165  PreallocatedType *cur = (PreallocatedType *) inst->pts.GetFirstItem();
166  while (cur != 0) {
167  cur->Cleanup();
168  cur = (PreallocatedType *) inst->pts.GetNextItem(cur);
169  }
170  printf("MEM: End cleanup.\n");
171 
172  pthread_mutex_unlock(&inst->mgr_thread_lock);
173 
174  return 0;
175 }

177 PreallocatedType::PreallocatedType(MemoryManager *mmgr,
178  Preallocated *prealloc_base,
179  int instance_size,
180  int prealloc_num_instances,
181  char block_mode) :
182  prealloc_base(prealloc_base), instance_size(instance_size),
183  prealloc_num_instances(prealloc_num_instances), block_mode(block_mode), blocks_list(0),
184  mmgr(mmgr) {
185  // Set up ready and free lists
186  ready_list = new RTStore<PreallocatedInstance>(prealloc_num_instances);
187 
188  // Setup base instance
189  prealloc_base->
190  SetupPreallocated(this,Preallocated::PREALLOC_BASE_INSTANCE);
191 
192  if (block_mode) {
193  // Setup one base block of instances
194  if (prealloc_num_instances < 3) {
195  printf("MEM: ERROR- Types using block mode must have at least 3 instances.");
196  exit(1);
197  }
198 
199  // Mark first instance in block as base
200  Preallocated *cur = prealloc_base;
201 
202  // Store the first block of instances
203  blocks_list = prealloc_base;
204 
205  /* printf("MEM: NEW BLOCK (STARTUP): %p, size: %d\n",cur,
206  instance_size); */
207 
208  // The rest are going right into the 'ready list'.. mark them and set them up
209  cur = (Preallocated *) (((char *) cur) + instance_size);
210  for (int i = 1; i < prealloc_num_instances; i++,
211  cur = (Preallocated *) (((char *) cur) + instance_size)) {
212  cur->SetupPreallocated(this,Preallocated::PREALLOC_IN_USE);
213  if (i == 1) // Second instance contains # of free instances
214  cur->predata.prealloc_num_free = 0; // None are free. We have the base instance and then the
215  // rest go straight to the ready list.
216 
217  // Add to ready list...
218  PreallocatedInstance *inst = 0;
219  int inst_idx; // Index in ready list
220  if ((inst = ready_list->FindItemWithState(RTStore<PreallocatedInstance>::ITEM_DONE,
221  RTStore<PreallocatedInstance>::ITEM_BUSY,inst_idx)) == 0) {
222  printf("MEM: ERROR: Ready_list size mismatch.\n");
223  exit(1);
224  } else
225  inst->ptr = cur;
226  if (!ready_list->ChangeStateAtIdx(inst_idx,RTStore<PreallocatedInstance>::ITEM_BUSY,
227  RTStore<PreallocatedInstance>::ITEM_WAITING)) {
228  printf("MEM: ERROR: State mismatch in ready_list\n");
229  exit(1);
230  }
231  }
232  } else {
233  // Instance mode
234 
235  // Create full ready list
236  PreallocatedInstance *i = 0;
237  do {
238  int idx;
239  if ((i = ready_list->FindItemWithState(RTStore<PreallocatedInstance>::ITEM_DONE,
240  RTStore<PreallocatedInstance>::ITEM_BUSY,idx)) != 0) {
241  // Prepare an instance and store in the list
242  i->ptr = prealloc_base->NewInstance();
243  i->ptr->SetupPreallocated(this,Preallocated::PREALLOC_IN_USE);
244 
245  // Update state to waiting (for consumption)
246  if (!ready_list->ChangeStateAtIdx(idx,RTStore<PreallocatedInstance>::ITEM_BUSY,
247  RTStore<PreallocatedInstance>::ITEM_WAITING)) {
248  printf("MEM: ERROR: State mismatch in ready_list\n");
249  exit(1);
250  }
251  }
252  } while (i != 0);
253  }
254 
255  // Add ourselves to list of managed types
256  mmgr->AddType(this);
257 };

267 Preallocated *PreallocatedType::RTNew() {
268  // Simply scan through the ready list and get an item marked 'waiting' (for consumption)
269  int idx;
270  PreallocatedInstance *i = ready_list->FindItemWithState(RTStore<PreallocatedInstance>::ITEM_WAITING,
271  RTStore<PreallocatedInstance>::ITEM_BUSY,idx);
272  if (i != 0) {
273  // Now, grab the pointer
274  Preallocated *ptr = i->ptr;
275 
276  // And mark it as 'done' (in use)
277  if (!ready_list->ChangeStateAtIdx(idx,RTStore<PreallocatedInstance>::ITEM_BUSY,
278  RTStore<PreallocatedInstance>::ITEM_DONE)) {
279  printf("MEM: ERROR: State mismatch in ready_list\n");
280  exit(1);
281  }
282 
283  // Wake up manager thread to replace the consumed instance
284  MemoryManagerUpdate mmu(this,T_MU_RestockInstance,idx,0);
285  mmgr->WakeUp(mmu);
286 
287  return ptr;
288  } else {
289  // No instances ready for consumption
290  printf("\nMEM: RTNew- No instances available.\n");
291  // mmgr->WakeupIfNeeded(1);
292  return 0;
293  }

316 void PreallocatedType::GoPostdelete(Preallocated *tofree) {
317  // *** NOTE: Currently no support for deleting additional blocks
318  // of instances- so if we are bursting to many instances
319  // the memory overhead will remain ***
320  //
321  // This should not use significant memory because, in Freewheeling, the large data structures
322  // such as AudioBlock are allocated in instance mode, not block mode.
323 
324  if (block_mode) {
325  // Recycle the instance (instead of deleting)
326  tofree->Recycle();
327 
328  // Mark as free
329  tofree->prealloc_status = Preallocated::PREALLOC_FREE;
330 
331  // Check pointer range to see which block this instance is in
332  Preallocated *cur = blocks_list;
333  int blocksize = instance_size*prealloc_num_instances;
334  while (cur != 0 && (tofree < cur || (char *) tofree >= ((char *) cur) + blocksize))
335  cur = cur->predata.prealloc_next;
336  if (cur == 0) {
337  printf("MEM: ERROR: Postdelete: Instance not found in any blocks\n");
338  exit(1);
339  } else {
340  // Add to free count
341  Preallocated *second = (Preallocated *) (((char *) cur) + instance_size);
342  second->predata.prealloc_num_free++;
343  }
344  } else {
345  // Single instance, simply delete it - no internal data structures to modify
346  ::delete tofree;
347  }

350 void PreallocatedType::GoPreallocate(int ready_list_idx) {
351  Preallocated *nw = 0;
352 
353  if (block_mode) {
354  // In block mode, we get a new index from our blocks list.
355  // Find a block with a free instance.
356  Preallocated *curblk = blocks_list;
357  char found = 0;
358  while (!found && curblk != 0) {
359  // Second instance in block contains # of free instances
360  Preallocated *second = (Preallocated *) (((char *) curblk) +
361  instance_size);
362  if (second->predata.prealloc_num_free > 0) {
363  // This block has a free instance. Use it.
364  found = 1;
365  } else
366  curblk = curblk->predata.prealloc_next;
367  }
368 
369  if (found) {
370  // Find a free instance in this block
371  Preallocated *curinst = curblk;
372  int i = 0;
373  for (; i < prealloc_num_instances && curinst->prealloc_status != Preallocated::PREALLOC_FREE;
374  i++, curinst = (Preallocated *) (((char *) curinst) + instance_size));
375  if (i >= prealloc_num_instances) {
376  // Failed, even though this block says it has free instances
377  printf("MEM: ERROR: Block marked with free instances but has none.\n");
378  exit(1);
379  } else {
380  // Use this instance
381  nw = curinst;
382  nw->prealloc_status = Preallocated::PREALLOC_IN_USE;
383 
384  // Reduce free count
385  Preallocated *second = (Preallocated *) (((char *) curblk) +
386  instance_size);
387  second->predata.prealloc_num_free--;
388  }
389  } else {
390  // No blocks found with free instances. We need a new block
391  // printf("MEM: New block allocated\n");
392  nw = prealloc_base->NewInstance();
393 
394  // Setup new block
395  Preallocated *cur = nw;
396  for (int i = 0; i < prealloc_num_instances; i++,
397  cur = (Preallocated *) (((char *) cur) + instance_size)) {
398  if (i == 0)
399  cur->SetupPreallocated(this,Preallocated::PREALLOC_IN_USE); // Use first instance
400  else
401  cur->SetupPreallocated(this,Preallocated::PREALLOC_FREE);
402 
403  if (i == 1) // Second instance contains # of free instances
404  // All but 1 (which we'll use) will be free
405  cur->predata.prealloc_num_free = prealloc_num_instances-1;
406  }
407 
408  // Link it in
409  nw->predata.prealloc_next = blocks_list;
410  blocks_list = nw;
411  }
412  } else {
413  // Single instance mode. Easy. Since one instance was consumed, we must allocate one more and
414  // put it in the ready list.
415  nw = prealloc_base->NewInstance();
416  nw->SetupPreallocated(this,Preallocated::PREALLOC_IN_USE);
417  }
418 
419  // Now place this instance in the ready list
420  if (nw != 0) {
421  if (!ready_list->ChangeStateAtIdx(ready_list_idx,RTStore<PreallocatedInstance>::ITEM_DONE,
422  RTStore<PreallocatedInstance>::ITEM_BUSY)) {
423  printf("MEM: ERROR: State mismatch in ready_list\n");
424  exit(1);
425  }
426 
427  ready_list->GetItemAtIdx(ready_list_idx)->ptr = nw;
428 
429  // Mark as waiting-for-consumption
430  if (!ready_list->ChangeStateAtIdx(ready_list_idx,RTStore<PreallocatedInstance>::ITEM_BUSY,
431  RTStore<PreallocatedInstance>::ITEM_WAITING)) {
432  printf("MEM: ERROR: State mismatch in ready_list\n");
433  exit(1);
434  }
435  } else {
436  printf("MEM: ERROR: Can't allocate more instances.\n");
437  exit(1);
438  }
439 };

444

453  curinst = (Preallocated *) (((char *) curinst) + instance_size)) {
