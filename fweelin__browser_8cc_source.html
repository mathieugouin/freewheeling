59 char ItemRenamer::HookEvent(Event *ev, EventProducer */*from*/) {
60  switch (ev->GetType()) {
61  case T_EV_Input_Key :
62  {
63  KeyInputEvent *kev = (KeyInputEvent *) ev;
64 
65  // Intercept alphanumeric keys (use UNICODE translation)
66  if ((kev->unicode > 32 &&
67  kev->unicode < 127) ||
68  (kev->unicode > 160 &&
69  kev->unicode < 256)) {
70  if (kev->down)
71  Rename_Append((char) (kev->unicode & 0xFF));
72  return 1;
73  } else if (kev->keysym == SDLK_BACKSPACE) {
74  if (kev->down)
75  Rename_Backspace();
76  return 1;
77  } else if (kev->keysym == SDLK_SPACE) {
78  if (kev->down)
79  Rename_Append(' ');
80 
81  } else if (kev->keysym == SDLK_RETURN ||
82  kev->keysym == SDLK_KP_ENTER) {
83  if (kev->down) {
84  app->getCFG()->RemoveEventHook(this);
85  app->getSDLIO()->EnableUNICODE(0);
86  app->getSDLIO()->EnableKeyRepeat(0);
87 
88  // Notify callback that item has been renamed
89  cb->ItemRenamed(rename_tmpbuf); // Callback may delete -this-
90  }
91  return 1;
92  } else if (kev->keysym == SDLK_ESCAPE) {
93  if (kev->down) {
94  app->getCFG()->RemoveEventHook(this);
95  app->getSDLIO()->EnableUNICODE(0);
96  app->getSDLIO()->EnableKeyRepeat(0);
97 
98  // Stop
99  cb->ItemRenamed(0); // Callback may delete -this-
100  }
101  return 1;
102  }
103 
104  return 0;
105  }
106  break;
107 
108  default:
109  break;
110  }
111 
112  return 0;

115 // Add an item to this browser
116 // (Doubley linked list add with sort)
117 // Nonzero if we should sort (according to the BrowserItem::Compare() method)
118 void Browser::AddItem(BrowserItem *nw, char sort) {
119  LockBrowser();
120 
121  if (first == 0) {
122  first = nw;
123  cur = first;
124  // printf("BROWSER: (id: %d) First item now: %s\n",id,first->name);
125  } else {
126  BrowserItem *cur = first;
127  if (sort && nw->Compare(cur) < 0) {
128  // Insert before first if that's where she goes
129  nw->next = first;
130  nw->prev = 0;
131  first->prev = nw;
132  first = nw;
133  } else {
134  // Insert at end or in the right place if sorting
135  if (sort) {
136  while (cur->next != 0 && nw->Compare(cur->next) >= 0)
137  cur = cur->next;
138  } else {
139  while (cur->next != 0)
140  cur = cur->next;
141  }
142 
143  // Insert after cur
144  nw->next = cur->next;
145  if (cur->next != 0)
146  cur->next->prev = nw;
147  nw->prev = cur;
148  cur->next = nw;
149  }
150  }
151 
152  UnlockBrowser();
153 };

155 // Remove an item from this browser
156 // For each item we call the MatchItem(itemmatch) method in BrowserItem
157 // until MatchItem returns zero. Then we remove that item.
158 void Browser::RemoveItem(int itemmatch) {
159  LockBrowser();
160 
161  BrowserItem *curi = first;
162  while (curi != 0 && curi->MatchItem(itemmatch))
163  curi = curi->next;
164 
165  if (curi != 0) {
166  // Found, remove
167  if (curi->prev != 0)
168  curi->prev->next = curi->next;
169  else
170  first = curi->next;
171 
172  if (curi->next != 0)
173  curi->next->prev = curi->prev;
174 
175  // Move currently selected item if we just removed it
176  if (cur == curi)
177  cur = (curi->prev != 0 ? curi->prev : curi->next);
178 
179  delete curi;
180  }
181 
182  UnlockBrowser();
183 };

224 void Browser::ItemRenamedOnDisk(char *old_filename, char *new_filename,

227  new_filename == 0)

233  char *filename = 0;

236  else if (cur->GetType() == B_Scene)
237  filename = ((SceneBrowserItem *) cur)->filename;
238 
239  // printf("looking for oldfilename: %s .. cur: %s\n",
240  // old_filename, filename);

242  found = 1;
243  else
244  cur = cur->next;
245  }
246 
247  if (found) {
248  // Found this item in our browser-- rename in our browser
249 
250  // Assign new filename
251  char *filename = 0;
252  if (cur->GetType() == B_Loop)
253  filename = ((LoopBrowserItem *) cur)->filename;
254  else if (cur->GetType() == B_Scene)
255  filename = ((SceneBrowserItem *) cur)->filename;
256 
257  delete[] filename;
258  filename = new char[strlen(new_filename)+1];
259  strcpy(filename,new_filename);
260 
261  if (cur->GetType() == B_Loop)
262  ((LoopBrowserItem *) cur)->filename = filename;
263  else if (cur->GetType() == B_Scene)
264  ((SceneBrowserItem *) cur)->filename = filename;
265 
266  // Assign new name and stop
267  RenameItem(cur,new_name);
268 
269  // If new name is blank, assign default name for onscreen display
270  if (cur->name == 0 || strlen(cur->name) == 0) {

290 // Get the onscreen display name for a file with given name.
291 // (filename must refer to a file of the type this browser handles)
292 // Write the name to outbuf, with max maxlen characters.
293 // Returns nonzero if we used a 'default' name.
294 char Browser::GetDisplayName(char *filename,
295  time_t *filetime,
296  char *outbuf, int maxlen) {
297  // Loop exists, use combination of time and hash as name
298  int baselen = strlen(app->getCFG()->GetLibraryPath()) + 1;
299  if (btype == B_Loop)
300  baselen += strlen(FWEELIN_OUTPUT_LOOP_NAME);
301  else if (btype == B_Scene)
302  baselen += strlen(FWEELIN_OUTPUT_SCENE_NAME);
303  else {
304  printf("BROWSER: We don't support getting a display name for type %d\n",
305  btype);
306  return 1;
307  }
308 
309  char sf_basename[FWEELIN_OUTNAME_LEN],
310  sf_hash[FWEELIN_OUTNAME_LEN],
311  sf_objname[FWEELIN_OUTNAME_LEN];
312 
313  if (Saveable::SplitFilename(filename, baselen, sf_basename, sf_hash,
314  sf_objname,FWEELIN_OUTNAME_LEN) ||
315  strlen(sf_objname) == 0) {
316  // No object name given in filename
317  // Compute default name
318  char hashshort_1 = 'X',
319  hashshort_2 = 'X';
320  Saveable::GetHashFirst(filename,baselen,&hashshort_1,&hashshort_2);
321 
322  // Compose name for item
323  snprintf(outbuf,maxlen,"%c%c %s",hashshort_1,hashshort_2,ctime(filetime));
324  outbuf[maxlen-1] = '\0';
325  outbuf[strlen(outbuf)-1] = '\0'; // Cut off return character
326 
327  return 1;
328  } else {
329  // Use name given in filename
330  strncpy(outbuf,sf_objname,maxlen);
331  outbuf[maxlen-1] = '\0';
332 
333  return 0;
334  }
335 }

372 // Add divisions between browser items using the Compare function-
373 // whenever two neighbouring items have difference greater than maxdelta,
374 // a division is inserted. In browsing files, for example, this allows us
375 // to group files that where created close to one another in time
376 //
377 // We also put divisions around any items that have been given a unique
378 // name
379 void Browser::AddDivisions(int maxdelta) {
380  if (first != 0) {
381  BrowserItem *cur = first;
382  char go = 1;
383  do {
384  while (cur->next != 0 && cur->next->Compare(cur) < maxdelta &&
385  ((cur->default_name && cur->next->default_name) ||
386  cur->GetType() == B_Division ||
387  cur->next->GetType() == B_Division))
388  cur = cur->next;
389 
390  if (cur->next != 0) {
391  // Add a division between cur and cur->next
392  BrowserItem *div = new BrowserDivision();
393  div->prev = cur;
394  div->next = cur->next;
395  cur->next->prev = div;
396  cur->next = div;
397 
398  cur = div->next;
399  } else
400  // Done
401  go = 0;
402  } while (go);
403  }

406 void Browser::MoveTo(int adjust, int jumpadjust) {
407  LockBrowser();
408 
409  if (xpand) {
410  xpand_lastactivity = mygettime();
411  xpanded = 1;
412  }
413 
414  if (cur == 0)
415  cur = first;
416 
417  if (cur != 0) {
418  // Jump by jumpadjust
419  int adjdir = (jumpadjust >= 0 ? 1 : -1),
420  adjmag = (adjdir == 1 ? abs(jumpadjust) : abs(jumpadjust)+1);
421  BrowserItem *prev = cur;
422  for (int i = 0; cur != 0 && i < adjmag; i++) {
423  prev = cur;
424 
425  while (cur != 0 && cur->GetType() != B_Division)
426  cur = (adjdir == 1 ? cur->next : cur->prev);
427  if (cur != 0)
428  // We are on the division, skip over it which way?
429  cur = (adjdir == 1 || i+1 >= adjmag ? cur->next : cur->prev);
430  if (cur == 0) {
431  if (adjdir == -1 && i+1 >= adjmag) {
432  // Going back to beginning of list
433  cur = first;
434  } else
435  // Went too far, go back
436  cur = prev;
437  }
438  }
439 
440  // Move by adjust
441  adjdir = (adjust >= 0 ? 1 : -1);
442  adjmag = abs(adjust);
443  for (int i = 0; cur != 0 && i < adjmag; i++) {
444  prev = cur;
445  cur = (adjdir == 1 ? cur->next : cur->prev);
446  if (cur != 0 && cur->GetType() == B_Division)
447  i--; // Don't count divisions
448  }
449  if (cur == 0)
450  cur = prev;
451  } else {
452  printf("BROWSER: No elements to move to for browser '%s'\n",
453  GetTypeName(GetType()));
454  }
455 
456  // Call browser virtual method for item browsed-
457  // derived types of browsers may need to do internal updates
458  ItemBrowsed();
459 
460  // Send out 'item-browsed' event-- which can be bound to
461  // in the config file.
462  BrowserItemBrowsedEvent *bievt = (BrowserItemBrowsedEvent *)
463  Event::GetEventByType(T_EV_BrowserItemBrowsed);
464  bievt->browserid = id;
465  // printf("BROWSER: itembrowsed send %p id: %d\n",bievt,id);
466  app->getEMG()->BroadcastEventNow(bievt, this);
467 
468  if (cur != 0 && callback != 0)
469  callback->ItemBrowsed(cur);
470 
471  UnlockBrowser();
472 };

577 void LoopTray::ReceiveEvent(Event *ev, EventProducer *from) {
578  switch (ev->GetType()) {
579  case T_EV_TriggerSet :
580  {
581  TriggerSetEvent *tev = (TriggerSetEvent *) ev;
582  // printf("tmap set: %d->%p\n",tev->idx,tev->nw);
583 
584  if (tev->nw == 0) {
585  // Delete,
586  // Scan for a LoopTrayItem corresponding to this index
587  RemoveItem(tev->idx);
588  touchtray = 1;
589  } else {
590  // Get the placename- where the loop is mapped
591  char *pname = 0;
592  FloLayout *curl = app->getCFG()->GetLayouts();
593  while (curl != 0) {
594  if (tev->idx >= curl->loopids.lo &&
595  tev->idx <= curl->loopids.hi) {
596  // Loop is mapped to an element in this layout- get the name
597  int firstid = curl->loopids.lo;
598  FloLayoutElement *curel = curl->elems;
599  while (curel != 0 && firstid + curel->id != tev->idx)
600  curel = curel->next;
601 
602  if (curel != 0)
603  pname = curel->name;
604  }
605 
606  curl = curl->next;
607  }
608 
609  // Add,
610  LoopTrayItem *nw = new LoopTrayItem(tev->nw,tev->idx,
611  tev->nw->name,
612  (tev->nw->name == 0 ? 1 : 0),
613  pname);
614  AddItem(nw,1);
615  touchtray = 1;
616  }
617  }
618  break;
619 
620  case T_EV_RenameLoop :
621  {
622  RenameLoopEvent *rev = (RenameLoopEvent *) ev;
623 
624  if (rev->in == 0) {
625  Rename(rev->loopid);
626  if (CRITTERS)
627  printf("LOOPTRAY: Received RenameLoop(loopid: %d)\n",rev->loopid);
628  }
629  }
630  break;
631 
632  default:
633  // Let base class handle it
634  Browser::ReceiveEvent(ev,from);
635  break;
636  }

705 char LoopTray::MouseMotion(MouseMotionInputEvent *mev) {
706  if (resize_win != RS_Off) {
707  int xdelta = mev->x - resize_xhand,
708  ydelta = mev->y - resize_yhand;
709 
710  if (resize_win == RS_Move) {
711  xpand_x1 = old_xpand_x1 + xdelta;
712  xpand_y1 = old_xpand_y1 + ydelta;
713  xpand_x2 = old_xpand_x2 + xdelta;
714  xpand_y2 = old_xpand_y2 + ydelta;
715  } else {
716  if ((resize_win == RS_Right || resize_win == RS_TopRight ||
717  resize_win == RS_BottomRight) &&
