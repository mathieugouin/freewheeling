26 /*
27  * This source implements memory manager classes that offer very fast, lightweight, realtime and
28  * thread-safe allocation and deallocation of arbitrary class instances. Using this approach, instances
29  * are preallocated and postdeleted and stored in lock-free real-time data structures.
30  *
31  * This allows real-time and time critical threads to get new instances on demand, and to free them without
32  * pausing. The actual memory allocation and deletion are managed in a manager thread.
33  */

113 // This class is a base for classes that want to be preallocated
114 // and postdeleted-- classes in which new instances are needed
115 // in realtime-- the memory allocation is done in a nonrealtime thread
116 //
117 // One such allocate&delete thread exists for all preallocated types
118 // MemoryManager handles that thread.
119 //
120 // Using this class as a base adds some bytes of size to each instance
121 // which might be a concern if you are allocating many instances!

307  // until it's woken again. So, set a flag and the RT audio
308  // thread will wake it up next process cycle.
309 
310  if (always_wakeup) {
311  // printf("MEM: WARNING: Priority inversion while filling update queue!\n");
312  needs_wakeup = 1;
313  }
314  }
315  }
