351  // These methods add and remove sync positions.
352  // A pulse sends out an RT PulseSync event whenever any of these
353  // positions is reached. (RT Safe, but can only be called from one thread- RT audio thread)
354  inline int AddPulseSync(PulseSyncCallback *cb, nframes_t pos) { // Returns sync index of new position
355  // Check position
356  if (pos >= len) {
357  printf("PULSE: Sync position adjusted for really short loop "
358  "(pos %d, len %d)\n",pos,len);
359  pos = len-1;
360  }
361 
362  // First, search for unfilled positions in our array
363  int i = 0;
364  while (i < numsyncpos && syncpos[i].cb != 0)
365  i++;
366  if (i < numsyncpos) {
367  // Position found, use this index
368  syncpos[i] = PulseSync(cb,pos);
369  return i;
370  } else {
371  // No holes found, add to end of array
372  if (numsyncpos >= MAX_SYNC_POS) {
373  printf("PULSE: Too many sync positions.\n");
374  return -1;
375  } else {
376  int ret = numsyncpos;
377  syncpos[numsyncpos++] = PulseSync(cb,pos);
378  return ret;
379  }
380  }
381  };

383  // Removes sync position at index syncidx (RT Safe, but can only be called from one thread- RT audio thread)
384  inline void DelPulseSync (int syncidx) {
385  if (syncidx < 0 || syncidx >= numsyncpos)
386  printf("PULSE: Invalid sync position %d (0->%d).\n",syncidx,numsyncpos);
387  else {
388  if (syncidx+1 == numsyncpos) {
389  // Position exists on the end of array- shrink
390  syncpos[syncidx] = PulseSync();
391  numsyncpos--;
392  } else
393  // Position exists in the middle of the array- create a hole
