47 int AudioIO::process (nframes_t nframes, void *arg) {
48  AudioIO *inst = static_cast<AudioIO *>(arg);
49 
50  if (inst->audio_thread == 0)
51  inst->audio_thread = pthread_self();
52 
53  // Check if EMG or MEM needs wakeup
54  inst->app->getEMG()->WakeupIfNeeded();
55  inst->app->getMMG()->WakeupIfNeeded();
56 
57  // Get CPU load
58  inst->cpuload = jack_cpu_load(inst->client);
59 
60  // Get JACK transport timing and status
61  int tmp_roll =
62  (jack_transport_query(inst->client,&(inst->jpos)) == JackTransportRolling);
63  if (inst->jpos.valid & JackPositionBBT)
64  inst->sync_active = 1;
65  else
66  inst->sync_active = 0;
67 
68  // Get buffers from jack
69  AudioBuffers *ab = inst->app->getABUFS();
70  for (int i = 0; i < ab->numins_ext; i++) {
71  // Left/mono channel
72  ab->ins[0][i] =
73  (sample_t *) jack_port_get_buffer (inst->iport[0][i], nframes);
74  // Right channel
75  ab->ins[1][i] = (inst->iport[1][i] != 0 ?
76  (sample_t *)
77  jack_port_get_buffer (inst->iport[1][i], nframes) :
78  0);
79  }
80  for (int i = 0; i < ab->numouts; i++) {
81  // Left/mono channel
82  ab->outs[0][i] =
83  (sample_t *) jack_port_get_buffer (inst->oport[0][i], nframes);
84  // Right channel
85  ab->outs[1][i] = (inst->oport[1][i] != 0 ?
86  (sample_t *)
87  jack_port_get_buffer (inst->oport[1][i], nframes) :
88  0);
89  }
90 
91  if (inst->rp != 0) {
92  if (nframes != inst->app->getBUFSZ()) {
93  printf("AUDIO: We've got a problem, honey!--\n");
94  printf("Audio buffer size has changed: %d->%d\n",
95  inst->app->getBUFSZ(),nframes);
96  exit(1);
97  }
98 
99  // Run through audio processors
100  inst->rp->process(0, nframes, ab);
101  }
102 
103  inst->timebase_master = 0; // Reset timebase master flag-
104  // callback will set to 1 if we are the master
105  inst->transport_roll = tmp_roll; // Set transport rolling status
106  return 0;
107 }

118 void AudioIO::timebase_callback(jack_transport_state_t /*state*/,
119  jack_nframes_t /*nframes*/,
120  jack_position_t *pos, int new_pos, void *arg) {
121  AudioIO *inst = static_cast<AudioIO *>(arg);
122 
123  // Set timebase master flag
124  inst->timebase_master = 1;
125 
126  /* printf("timebase called back (frame: %d, framerate: %d)!\n",
127  pos->frame,pos->frame_rate); */
128 
129  // Use our pulse information plus JACK's frame information
130  // to calculate bars and beats and ticks
131  Pulse *p = inst->app->getLOOPMGR()->GetCurPulse();
132  if (p != 0) {
133  if (new_pos) {
134  if (inst->repos)
135  inst->repos = 0; // JACK telling us we have moved- but we initiated
136  // the move, so ignore
137  else {
138  // Somebody has started the transport at a new position-
139  // signal back that we want to start at a new position--
140  // based on the current pulse position.
141 
142  // printf("relocate: posframe: %d to %d\n",pos->frame,p->GetPos());
143  jack_transport_locate(inst->client,p->GetPos());
144  inst->repos = 1;
145 
146 #if 0
147  inst->sync_start_frame = pos->frame + (p->GetLength()-p->GetPos());
148  printf("posframe: %d syncstartframe set: %d\n",pos->frame,
149  inst->sync_start_frame);
150 #endif
151 
152  inst->sync_start_frame = 0; //pos->frame;
153  }
154  }
155 
156 #define TICKS_PER_BEAT 1920
157 
158  int32_t rel_frame = (int) pos->frame - inst->sync_start_frame;
159  float rel_bar = (float) rel_frame*inst->app->GetSyncSpeed()/p->GetLength();
160  if (inst->app->GetSyncType() != 0)
161  // Beat sync, adjust bar by SYNC_BEATS_PER_BAR
162  rel_bar /= SYNC_BEATS_PER_BAR;
163  pos->valid = JackPositionBBT;
164  pos->bar = (int32_t) rel_bar;
165  float bar_frac = rel_bar - pos->bar;
166  pos->beat = (int) (bar_frac * SYNC_BEATS_PER_BAR);
167  pos->beats_per_bar = SYNC_BEATS_PER_BAR;
168  if (inst->app->GetSyncType() == 0)
169  pos->beats_per_minute = SYNC_BEATS_PER_BAR*
170  60.0*(double) pos->frame_rate/p->GetLength();
171  else
172  pos->beats_per_minute = 60.0*(double) pos->frame_rate/p->GetLength();
173  pos->beat_type = SYNC_BEATS_PER_BAR;
174  pos->ticks_per_beat = TICKS_PER_BEAT;
175  float beat_frac = bar_frac*SYNC_BEATS_PER_BAR - (float)pos->beat;
176  pos->tick = (int) (beat_frac * TICKS_PER_BEAT);
177 
178  pos->bar++;
179  pos->beat++;
180  //pos->tick++;
181 
182  pos->bar_start_tick = pos->bar * SYNC_BEATS_PER_BAR * TICKS_PER_BEAT;
183 
184  //printf("ticks per beat: %f\n",pos->ticks_per_beat);
185  //printf("rel bar: %f, bar: %d, beat: %d, tick: %d\n",rel_bar, pos->bar,
186  // pos->beat, pos->tick);
187  }
188 }

273 int AudioIO::open () {
274  // **** AUDIO startup
275 
276  // Try to become a client of the JACK server
277  client = jack_client_open("FreeWheeling", JackNoStartServer, NULL);
278  if (!client) {
279  fprintf (stderr, "AUDIO: ERROR: Jack server not running!\n");
280  return 1;
281  }
282 
283  /* tell the JACK server to call `process()' whenever
284  there is work to be done.
285  */
286  jack_set_process_callback (client, process, this);
287 
288  jack_nframes_t bufsz = jack_get_buffer_size (client);
289  printf ("AUDIO: Audio buffer size is: %d\n", bufsz);
290 
291  /* tell the JACK server to call `srate_callback()' whenever
292  the sample rate of the system changes.
293  */
294  jack_set_sample_rate_callback (client, srate_callback, this);
295 
296  /* tell the JACK server to call `jack_shutdown()' if
297  it ever shuts down, either entirely, or if it
298  just decides to stop calling us.
299  */
300  jack_on_shutdown (client, audio_shutdown, this);
301 
302  // Set timebase callback
303  jack_set_timebase_callback (client, 1, timebase_callback, this);
304 
305  /* display the current sample rate. once the client is activated
306  (see below), you should rely on your own sample rate
307  callback (see above) for this value.
308  */
309  srate = jack_get_sample_rate (client);
310  printf ("AUDIO: Engine sample rate is %d\n", srate);
311 
312  // Set time scale
313  timescale = (float) bufsz/srate;
314 
315  repos = 0;
316 
317  // Create buffers
318  AudioBuffers *ab = app->getABUFS();
319 
320  printf("AUDIO: Using %d external inputs, %d total inputs\n",
321  ab->numins_ext,ab->numins);
322  iport[0] = new jack_port_t *[ab->numins_ext];
323  iport[1] = new jack_port_t *[ab->numins_ext];
324  oport[0] = new jack_port_t *[ab->numouts];
325  oport[1] = new jack_port_t *[ab->numouts];
326 
327  // Create ports
328  char tmp[255];
329  for (int i = 0; i < ab->numins_ext; i++) {
330  char stereo = ab->IsStereoInput(i);
331  snprintf(tmp,255,"in_%d%s",i+1,(stereo ? "L" : ""));
332  iport[0][i] = jack_port_register(client, tmp, JACK_DEFAULT_AUDIO_TYPE,
333  JackPortIsInput, 0);
334  if (stereo) {
335  snprintf(tmp,255,"in_%d%s",i+1,"R");
336  iport[1][i] = jack_port_register(client, tmp, JACK_DEFAULT_AUDIO_TYPE,
337  JackPortIsInput, 0);
338  } else
339  iport[1][i] = 0;
340  }
341 
342  for (int i = 0; i < ab->numouts; i++) {
343  char stereo = ab->IsStereoOutput(i);
344  snprintf(tmp,255,"out_%d%s",i+1,(stereo ? "L" : ""));
345  oport[0][i] = jack_port_register(client, tmp, JACK_DEFAULT_AUDIO_TYPE,
346  JackPortIsOutput, 0);
347  if (stereo) {
348  snprintf(tmp,255,"out_%d%s",i+1,"R");
349  oport[1][i] = jack_port_register(client, tmp, JACK_DEFAULT_AUDIO_TYPE,
350  JackPortIsOutput, 0);
351  } else
352  oport[1][i] = 0;
353  }
354 
355  return 0;
356 }
