56 // Event with normal (direct method call) delivery using block allocation with
57 // default number of instances preallocated
58 #define SET_ETYPE(etyp,nm,typ) \
59  case etyp : \
60  { \
61  Event *proto = \
62  ::new typ[PreallocatedType:: \
63  PREALLOC_DEFAULT_NUM_INSTANCES]; \
64  ett[i].name = nm; \
65  ett[i].pretype = \
66  new PreallocatedType(mmgr,proto,sizeof(typ), \
67  PreallocatedType:: \
68  PREALLOC_DEFAULT_NUM_INSTANCES, \
69  1); \
70  ett[i].slowdelivery = 0; \
71  int paramidx = -1, j = 0; \
72  for (; j < proto->GetNumParams() && \
73  proto->GetParam(j).max_index == -1; j++); \
74  if (j < proto->GetNumParams()) \
75  paramidx = j; \
76  ett[i].paramidx = paramidx; \
77  } \
78  break;

347  int evnum = (int) EventType(T_EV_Last);
348  for (int i = 0; i < evnum; i++) {
349  // Deleting the manager will delete all instances

357  ett[i].proto = 0;
358  }

363 Event *Event::GetEventByType(EventType typ, char wait) {
364  if (ett == 0) {
365  printf("EVENT: ERROR- no event type table!\n");
366  exit(1);
367  }
368 
369  int i = (int) typ;
370  if (ett[i].pretype != 0) {
371  Event *ret = (Event *) ett[i].pretype->RTNew();
372  if (ret != 0)
373  return ret;
374  else {
375  // No instance available
376  if (wait) {
377  // Wait
378  printf("EVENT: Waiting for memory allocation of event '%s'.\n",ett[i].name);
379  do {
380  usleep(10000);
381  ret = (Event *) ett[i].pretype->RTNew();
382  } while (ret == 0);
383  return ret;
384  } else {
385  printf("EVENT: ERROR: No wait condition and no instances available for event '%s'\n",ett[i].name);
386  return 0; // Don't wait
387  }
388  }
389  }
390  else if (ett[i].proto != 0) {
391  Event *ret = (Event *) ett[i].proto->RTNew();
392  if (ret != 0)
393  return ret;
394  else {
395  // No instance available
396  if (wait) {
397  // Wait
398  printf("EVENT: Waiting for memory allocation of event '%s'.\n",ett[i].name);
399  do {
400  usleep(10000);
401  ret = (Event *) ett[i].proto->RTNew();
402  } while (ret == 0);
403  return ret;
404  } else {
405  printf("EVENT: ERROR: No wait condition and no instances available for event '%s'\n",ett[i].name);
406  return 0; // Don't wait
407  }
408  }
409  }
410  else {
411  printf("ERROR: no prototype or mgr for event type: '%s'\n",
412  ett[i].name);
413  return 0;
414  }

433 EventManager::EventManager () : eq(0), needs_wakeup(0), threadgo(1) {
434  printf("Start event manager.\n");
435 
436  // Create listener structure..
437  int evnum = (int) EventType(T_EV_Last);
438  listeners = new EventListenerItem *[evnum];
439  for (int i = 0; i < evnum; i++)
440  listeners[i] = 0;
441 
442  pthread_mutex_init(&dispatch_thread_lock,0);
443  pthread_mutex_init(&listener_list_lock,0);
444  pthread_cond_init(&dispatch_ready,0);
445 
446  const static size_t STACKSIZE = 1024*128;
447  pthread_attr_t attr;
448  pthread_attr_init(&attr);
449  pthread_attr_setstacksize(&attr,STACKSIZE);
450  printf("EVENT: Stacksize: %zd.\n",STACKSIZE);
451 
452  // Hold dispatch thread until ready
453  pthread_mutex_lock(&dispatch_thread_lock);
454 
455  // Start an event dispatch thread
456  int ret = pthread_create(&dispatch_thread,
457  &attr,
458  run_dispatch_thread,
459  static_cast<void *>(this));
460  if (ret != 0) {
461  printf("(eventmanager) pthread_create failed, exiting");
462  exit(1);
463  }
464  RT_RWThreads::RegisterReaderOrWriter(dispatch_thread);
465 
466  struct sched_param schp;
467  memset(&schp, 0, sizeof(schp));
468 
469  // Event dispatch thread is NOT RT
470  schp.sched_priority = sched_get_priority_max(SCHED_OTHER);
471  if (pthread_setschedparam(dispatch_thread, SCHED_OTHER, &schp) != 0) {
472  printf("EVENT: Can't set hi priority thread, will use regular!\n");
473  }
474 };

476 EventManager::~EventManager() {
477  //printf("Event Manager: cleanup... this: %p\n",this);
478 
479  // Terminate the dispatch thread
480  threadgo = 0;
481  pthread_mutex_lock (&dispatch_thread_lock);
482  pthread_cond_signal (&dispatch_ready);
483  pthread_mutex_unlock (&dispatch_thread_lock);
484  pthread_join(dispatch_thread,0);
485 
486  pthread_cond_destroy (&dispatch_ready);
487  pthread_mutex_destroy (&dispatch_thread_lock);
488  pthread_mutex_destroy (&listener_list_lock);
489 
490  int evnum = (int) EventType(T_EV_Last);
491  for (int i = 0; i < evnum; i++) {
492  // Erase listeners
493  EventListenerItem *cur = listeners[i];
494  while (cur != 0) {
495  EventListenerItem *tmp = cur->next;
496  delete cur;
497  cur = tmp;
498  }
499  }
500 
501  delete[] listeners;
502  if (eq != 0)
503  delete eq;
504 
505  // Takedown event type table
506  // printf(" .. ETT takedown (this: %p)\n",this);
507  Event::TakedownEventTypeTable();
508  // printf(" .. ETT takedown done (this: %p)\n",this);
509 
510  printf("EVENT: manager end.\n");
511 };

629 void EventManager::ListenEvent(EventListener *callme,
630  EventProducer *from, EventType type,
631  char block_self_calls) {

652 void EventManager::UnlistenEvent(EventListener *callme,
653  EventProducer *from, EventType type) {
654  pthread_mutex_lock(&listener_list_lock);
655 
656  // Remove from the listeners list
657  int evnum = (int) type;
658  EventListenerItem *cur = listeners[evnum],
659  *prev = 0;
660 
661  // Search for those listening to 'from' & 'type'
662  while (cur != 0 && (cur->callwhom != callme ||
663  cur->eventsfrom != from)) {
664  prev = cur;
665  cur = cur->next;
666  }
667 
668  if (cur != 0) {
669  // Got it, unlink!
670  if (prev != 0)
671  prev->next = cur->next;
672  else
673  listeners[evnum] = cur->next;
674  delete cur;
675  }
676 
677  pthread_mutex_unlock(&listener_list_lock);
678 };
