86 int SndFileDecoder::ReadSamples(AudioBlockIterator *it, nframes_t max_len) {
87  int len = 0;
88 
89  if (stereo) {
90  if (filetype == FLAC){ // need to convert from int24
91  len = sf_read_float(sfinfd,rbuf,sfinfo.channels*max_len);
92  len /= sfinfo.channels;
93  } else { // everything else is currently native float
94  len = sf_read_raw(sfinfd,rbuf,sfinfo.channels*max_len*samplesize);
95  len /= (sfinfo.channels *samplesize);
96  }
97  for (nframes_t i = 0; i < max_len; i++) { // deinterleave channels using brute strength
98  obuf[0][i] = rbuf[2*i];
99  obuf[1][i] = rbuf[2*i+1];
100  }
101  if (len > 0) {
102  it->PutFragment(obuf[0],obuf[1],len,1);
103  it->NextFragment();
104  }
105 
106  } else {
107  if (filetype == FLAC) { // need to convert from int24
108  len = sf_read_float(sfinfd,obuf[0],sfinfo.channels*max_len);
109  } else { // everything else is currently native float
110  len = sf_read_raw(sfinfd,obuf[0],sfinfo.channels*max_len*samplesize);
111  }
112  if (len > 0) {
113  it->PutFragment(obuf[0],0,len,1);
114  it->NextFragment();
115  }
116  }

327 // This analyzes and dumps any remaining frames out to file
328 long int VorbisEncoder::Encode() {
329  if (outfd != 0) {
330  long int outputsize = 0;
331 
332  // Now do the real analysis!! Meat n potatoes!
333  while(vorbis_analysis_blockout(&vd,&vb) == 1) {
334  /* analysis, assume we want to use bitrate management */
335  vorbis_analysis(&vb,NULL);
336  vorbis_bitrate_addblock(&vb);
337 
338  while(vorbis_bitrate_flushpacket(&vd,&op)) {
339  /* weld the packet into the bitstream */
340  ogg_stream_packetin(&os,&op);
341 
342  /* write out pages (if any) */
343  char eos = 0;
344  while (!eos) {
345  int result = ogg_stream_pageout(&os,&og);
346  if (result == 0)
347  break;
348  fwrite(og.header,1,og.header_len,outfd);
349  fwrite(og.body,1,og.body_len,outfd);
350 
351  // Add to the output size
352  outputsize += og.header_len;
353  outputsize += og.body_len;
354 
355  if (ogg_page_eos(&og))
356  eos = 1;
357  }
358  }
359  }
360 
361  return outputsize;
362  } else {
363  printf("BLOCK: ERROR: OGG encoder finished but Encode() called!\n");
364  return 0;
365  }

523 int BlockReadManager::Manage() {
524  if (in == 0) {
525  // Not currently decoding
526  if (arc != 0) {
527  // We have a callback, get a chain to load
528  arc->GetReadBlock(&in,&smooth_end);
529 
530  if (in != 0)
531  // We got a chain to load, so begin
532  Start();
533  } else {
534  // No callback, so we are done!
535  return 1;
536  }
537  }
538 
539  if (in != 0) {
540  // Make sure we have started on this blockchain
541  if (dec == 0)
542  Start();
543 
544  // Continue decoding
545  pthread_mutex_lock (&decode_lock);
546 
547  // Now that we have the lock, make sure we are still active
548  if (in != 0) {
549  for (int pass = 0; in != 0 && pass < NUM_DECODE_PASSES; pass++) {
550  // Make sure we have an extra block in our chain
551  if (i->GetCurBlock()->next == 0) {
552  AudioBlock *nw = (AudioBlock *) i->GetCurBlock()->RTNew();
553  if (nw == 0) {
554  // Pause decode until block is here
555  printf("BlockReadManager: Waiting for block.\n");
556  pthread_mutex_unlock (&decode_lock);
557  return 0;
558  }
559 
560  i->GetCurBlock()->Link(nw);
561  }
562 
563  // Decode samples
564  int len = dec->ReadSamples(i,DECODE_CHUNKSIZE);
565  if (len == 0) {
566  // EOF
567  pthread_mutex_unlock (&decode_lock);
568  End(0);
569  } else if (len < 0) {
570  // Stream error- just continue
571  printf("BlockReadManager: Stream read error!\n");
572  }
573  }
574 
575  }
576 
577  pthread_mutex_unlock (&decode_lock);
578  }
579 
580  return 0;

677 int BlockWriteManager::Manage() {
678  if (b == 0) {
679  // Not currently encoding
680  if (awc != 0) {
681  // We have a callback, get a chain to save
682  awc->GetWriteBlock(&out,&b,&i,&len);
683 
684  if (b != 0)
685  // We got a chain to save, so begin
686  Start();
687  } else {
688  // No callback, so we are done!
689  return 1;
690  }
691  }
692 
693  if (b != 0) {
694  // Make sure we have started on this blockchain
695  if (enc == 0)
696  Start();
697 
698  // Continue encoding
699  pthread_mutex_lock (&encode_lock);
700 
701  // Now that we have the lock, make sure we are still active
702  if (b != 0) {
703  nframes_t remaining = len-pos,
704  num = MIN(ENCODE_CHUNKSIZE,remaining);
705 
706  sample_t *ibuf[2];
707  if (enc->IsStereo()) {
708  // Stereo
709  ei->GetFragment(&ibuf[0],&ibuf[1]);
