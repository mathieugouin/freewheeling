305 void Browser::Draw(SDL_Surface *screen) {
306  LockBrowser();
307 
308  if (xpanded) {
309  // Draw expanded view
310 
311  // Dim the background
312  for (int i = xpand_y1; i <= xpand_y2; i++)
313  hlineRGBA(screen,xpand_x1,xpand_x2,i,0,0,0,200);
314  hlineRGBA(screen,xpand_x1,xpand_x2,xpand_y1,127,127,127,255);
315  hlineRGBA(screen,xpand_x1,xpand_x2,xpand_y2,127,127,127,255);
316  vlineRGBA(screen,xpand_x1,xpand_y1,xpand_y2,127,127,127,255);
317  vlineRGBA(screen,xpand_x2,xpand_y1,xpand_y2,127,127,127,255);
318 
319  if (cur != 0) {
320  BrowserItem *sp_1 = cur,
321  *sp_2 = cur;
322 
323  // Compute text height and center of expanded window-- once!
324  if (xpand_liney == -1) {
325  TTF_SizeText(font->font,VERSION,0,&xpand_liney);
326  xpand_centery = (xpand_y1+xpand_y2)/2;
327  xpand_spread = MIN(xpand_centery-xpand_y1,
328  xpand_y2-xpand_centery);
329  xpand_spread /= xpand_liney;
330 
331  // printf("compute xpand_liney: %d xpand_centery: %d\n",
332  // xpand_liney, xpand_centery);
333  }
334 
335  int ofs_1 = 0;
336  boxRGBA(screen,
337  xpand_x1,xpand_centery,
338  xpand_x2,xpand_centery+xpand_liney,
339  127,0,0,255);
340  Draw_Item(screen,sp_1,xpand_x1,
341  xpand_centery-xpand_liney*ofs_1);
342  while (ofs_1 < xpand_spread && sp_1->prev != 0) {
343  sp_1 = sp_1->prev;
344  ofs_1++;
345  Draw_Item(screen,sp_1,xpand_x1,
346  xpand_centery-xpand_liney*ofs_1);
347  }
348  int ofs_2 = 0;
349  while (ofs_2 < xpand_spread-1 && sp_2->next != 0) {
350  sp_2 = sp_2->next;
351  ofs_2++;
352  Draw_Item(screen,sp_2,xpand_x1,
353  xpand_centery+xpand_liney*ofs_2);
354  }
355  }
356 
357  if (mygettime()-xpand_lastactivity >= xpand_delay) {
358  // No activity, close expanded window
359  xpanded = 0;
360  }
361  }
362 
363  // Draw single-line view
364  Draw_Item(screen,cur,xpos,ypos);
365 
366  UnlockBrowser();
367 }

552
