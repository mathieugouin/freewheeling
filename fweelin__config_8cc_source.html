1328  char enable_keynames) {

1330  char buf[255]; // Copy buf
1331  CfgMathOperation *first = 0,
1332  *last = 0;
1333  signed int firstopidx = -1;
1334 
1335  //printf("parse expression: %s\n",str);
1336 
1337  // Create list of operators
1338  int i;
1339  for (i = 0; i < CfgMathOperation::numops; i++)
1340  opstr[i] = CfgMathOperation::operators[i];
1341  opstr[i] = '\0';
1342 
1343  // Find the first occurance of an operator in string str
1344  char *cur = strpbrk(str,opstr);
1345  if (cur != 0)
1346  firstopidx = (long)cur-(long)str;
1347  else
1348  firstopidx = -1;

1351  ParsedExpression *exp = new ParsedExpression();
1352  // Parse beginning token
1353  if (firstopidx == -1) {
1354  strncpy(buf,str,255);
1355  buf[254] = '\0';
1356  }
1357  else {

1359  memcpy(buf,str,len);

1373  else
1374  len = strlen(cur);
1375  if (len >= 255)
1376  len = 254;
1377  memcpy(buf,cur,len);
1378  buf[len] = '\0';

1382  nw->otype = op;
